**Module shape (strict spec, no creative extension):**

```javascript
'use strict';

// Size thresholds from E5.8 RESULTS.md:71 — means observed in facets-covered vs uncovered sessions.
const SESSION_SIZE_THRESHOLDS = Object.freeze({
  small_max: 5,    // deduped_message_count <= 5
  medium_max: 20,  // deduped_message_count <= 20
  // large: > 20
});

// Write-path clustering parameters per E5.8 Finding A + MEAS-DERIVED-03 text.
const DEFAULT_CLUSTER_WINDOW_SECONDS = 2;
const DEFAULT_MIN_CLUSTER_SIZE = 5;

/**
 * Classify a session by size bucket using deduped_message_count.
 * Accepts a session object shaped like sources/claude.cjs `session` records;
 * falls back to `user_message_count` if `jsonl_usage.deduped_message_count` absent.
 * Returns { bucket: 'small'|'medium'|'large'|'unknown', count: number|null }.
 */
function classifySessionSize(session) {
  const count = (session && session.jsonl_usage && typeof session.jsonl_usage.deduped_message_count === 'number')
    ? session.jsonl_usage.deduped_message_count
    : (session && typeof session.user_message_count === 'number' ? session.user_message_count : null);
  if (count === null) return { bucket: 'unknown', count: null };
  if (count <= SESSION_SIZE_THRESHOLDS.small_max) return { bucket: 'small', count };
  if (count <= SESSION_SIZE_THRESHOLDS.medium_max) return { bucket: 'medium', count };
  return { bucket: 'large', count };
}

/**
 * Group files (array of { path, mtime }) into mtime clusters.
 * Files whose mtimes fall within `window_seconds` of another file's mtime are in the same cluster.
 * A cluster with >= `min_cluster_size` files represents a bulk-write event.
 * Returns a Map<cluster_id, { files: string[], window_start: ISO, window_end: ISO, size: number }>
 * cluster_id format: `cluster_<epoch_ms_of_earliest_mtime>_<size>`.
 *
 * Algorithm (single-pass O(n log n)):
 * 1. Sort files by mtime ascending.
 * 2. Walk sorted list; start a new cluster when gap from previous file > window_seconds.
 * 3. Return only clusters with size >= min_cluster_size (singletons are 'single' write path).
 *
 * Caller uses `classifyWritePath(mtime, cluster_map)` to look up cluster membership.
 */
function clusterByMtime(files, opts = {}) {
  const window_seconds = opts.window_seconds || DEFAULT_CLUSTER_WINDOW_SECONDS;
  const min_cluster_size = opts.min_cluster_size || DEFAULT_MIN_CLUSTER_SIZE;
  const window_ms = window_seconds * 1000;
  // Input guard
  if (!Array.isArray(files) || files.length === 0) return new Map();
  // Filter valid + sort
  const sorted = files
    .filter(f => f && f.path && (typeof f.mtime === 'number' || f.mtime instanceof Date || typeof f.mtime === 'string'))
    .map(f => ({ path: f.path, mtime_ms: toMs(f.mtime) }))
    .filter(f => !isNaN(f.mtime_ms))
    .sort((a, b) => a.mtime_ms - b.mtime_ms);
  // Single-pass clustering
  const clusters = [];
  let current = null;
  for (const f of sorted) {
    if (current && f.mtime_ms - current.last_mtime_ms <= window_ms) {
      current.files.push(f.path);
      current.last_mtime_ms = f.mtime_ms;
    } else {
      if (current && current.files.length >= min_cluster_size) clusters.push(current);
      current = { files: [f.path], first_mtime_ms: f.mtime_ms, last_mtime_ms: f.mtime_ms };
    }
  }
  if (current && current.files.length >= min_cluster_size) clusters.push(current);
  // Build Map keyed by cluster_id, with reverse lookup for classifyWritePath
  const map = new Map();
  for (const c of clusters) {
    const cluster_id = `cluster_${c.first_mtime_ms}_${c.files.length}`;
    map.set(cluster_id, {
      cluster_id,
      files: c.files,
      file_set: new Set(c.files),
      window_start: new Date(c.first_mtime_ms).toISOString(),
      window_end: new Date(c.last_mtime_ms).toISOString(),
      size: c.files.length,
      window_seconds: (c.last_mtime_ms - c.first_mtime_ms) / 1000,
    });
  }
  return map;
}

function toMs(mtime) {
  if (typeof mtime === 'number') return mtime;
  if (mtime instanceof Date) return mtime.getTime();
  if (typeof mtime === 'string') return new Date(mtime).getTime();
  return NaN;
}

/**
 * Classify a single file's mtime/path against a cluster map.
 * Returns { write_path: 'bulk'|'single', cluster_id: string|null, cluster_size: number|null }.
 */
function classifyWritePath(filePath, cluster_map) {
  if (!cluster_map || cluster_map.size === 0) return { write_path: 'single', cluster_id: null, cluster_size: null };
  for (const [cluster_id, c] of cluster_map.entries()) {
    if (c.file_set.has(filePath)) {
      return { write_path: 'bulk', cluster_id, cluster_size: c.size };
    }
  }
  return { write_path: 'single', cluster_id: null, cluster_size: null };
}

/**
 * Compose the full stratification object required on every MEAS-DERIVED-02 feature.
 * {
 *   write_path: 'bulk' | 'single',
 *   facets_coverage_class: 'with' | 'without',
 *   size_bucket: 'small' | 'medium' | 'large' | 'unknown',
 *   cluster_id: string | null,
 *   user_message_count: number | null
 * }
 *
 * Inputs:
 *   session: the full session object
 *   cluster_map: result of clusterByMtime(all session-meta + facets files)
 *   session_meta_path: the file path to classify (typically session.session_meta.path)
 *   has_facet: boolean indicating facet presence
 */
function buildStratificationObject({ session, cluster_map, session_meta_path, has_facet }) {
  const size = classifySessionSize(session);
  const path_classification = session_meta_path
    ? classifyWritePath(session_meta_path, cluster_map)
    : { write_path: 'single', cluster_id: null, cluster_size: null };
  return {
    write_path: path_classification.write_path,
    facets_coverage_class: has_facet ? 'with' : 'without',
    size_bucket: size.bucket,
    cluster_id: path_classification.cluster_id,
    user_message_count: size.count,
  };
}

module.exports = {
  SESSION_SIZE_THRESHOLDS,
  DEFAULT_CLUSTER_WINDOW_SECONDS,
  DEFAULT_MIN_CLUSTER_SIZE,
  classifySessionSize,
  clusterByMtime,
  classifyWritePath,
  buildStratificationObject,
};
```

**Implementation notes / anti-patterns to avoid:**
- NO filesystem reads inside the helper — the caller (extractor) supplies the files array from `context.claude.sessions[].session_meta.{path, mtime}`.
- NO caching at module level — the helper must be pure so rebuild is deterministic per-run.
- NO dependencies on registry.cjs / query.cjs — keep the helper a leaf.
- Pre-aggregation violation (G-7): `clusterByMtime` groups for classification, not for aggregation. It MUST NOT sum counts or produce statistics; it produces cluster membership only.

const { describe, it, expect } = require('vitest');
const {
  classifySessionSize,
  clusterByMtime,
  classifyWritePath,
  buildStratificationObject,
  SESSION_SIZE_THRESHOLDS,
} = require('../../get-shit-done/bin/lib/measurement/stratify.cjs');

describe('classifySessionSize', () => {
  it('classifies small (<=5) from jsonl_usage.deduped_message_count', () => {
    expect(classifySessionSize({ jsonl_usage: { deduped_message_count: 3 } })).toEqual({ bucket: 'small', count: 3 });
    expect(classifySessionSize({ jsonl_usage: { deduped_message_count: 5 } })).toEqual({ bucket: 'small', count: 5 });
  });
  it('classifies medium (6-20)', () => {
    expect(classifySessionSize({ jsonl_usage: { deduped_message_count: 15 } })).toEqual({ bucket: 'medium', count: 15 });
    expect(classifySessionSize({ jsonl_usage: { deduped_message_count: 20 } })).toEqual({ bucket: 'medium', count: 20 });
  });
  it('classifies large (>20)', () => {
    expect(classifySessionSize({ jsonl_usage: { deduped_message_count: 50 } })).toEqual({ bucket: 'large', count: 50 });
  });
  it('falls back to user_message_count when jsonl_usage absent', () => {
    expect(classifySessionSize({ user_message_count: 8 })).toEqual({ bucket: 'medium', count: 8 });
  });
  it('returns unknown when no count available', () => {
    expect(classifySessionSize({})).toEqual({ bucket: 'unknown', count: null });
  });
});

describe('clusterByMtime', () => {
  it('groups >=5 files within 2s window as one cluster', () => {
    const base = Date.UTC(2026, 3, 16, 11, 6, 46);
    const files = Array.from({ length: 6 }, (_, i) => ({ path: `/f/${i}.md`, mtime: base + i * 200 }));
    const map = clusterByMtime(files);
    expect(map.size).toBe(1);
    const [c] = map.values();
    expect(c.size).toBe(6);
    expect(c.window_seconds).toBeLessThanOrEqual(2);
  });
  it('does not cluster <5 files even if within window', () => {
    const base = Date.now();
    const files = [
      { path: '/a.md', mtime: base },
      { path: '/b.md', mtime: base + 500 },
      { path: '/c.md', mtime: base + 1000 },
    ];
    const map = clusterByMtime(files);
    expect(map.size).toBe(0);
  });
  it('splits clusters when gap exceeds window', () => {
    const base = Date.UTC(2026, 3, 16, 11, 6, 46);
    const files = [
      ...Array.from({ length: 5 }, (_, i) => ({ path: `/a${i}.md`, mtime: base + i * 100 })),
      ...Array.from({ length: 5 }, (_, i) => ({ path: `/b${i}.md`, mtime: base + 10000 + i * 100 })),
    ];
    const map = clusterByMtime(files);
    expect(map.size).toBe(2);
  });
  it('handles empty input', () => {
    expect(clusterByMtime([]).size).toBe(0);
    expect(clusterByMtime(null).size).toBe(0);
  });
});

describe('classifyWritePath', () => {
  it('returns bulk when file is in a cluster', () => {
    const base = Date.now();
    const files = Array.from({ length: 5 }, (_, i) => ({ path: `/x${i}.md`, mtime: base + i * 100 }));
    const map = clusterByMtime(files);
    expect(classifyWritePath('/x0.md', map).write_path).toBe('bulk');
  });
  it('returns single when file is not in any cluster', () => {
    const map = clusterByMtime([]);
    expect(classifyWritePath('/isolated.md', map).write_path).toBe('single');
  });
});

describe('buildStratificationObject', () => {
  it('composes all four axes from session + cluster_map + has_facet', () => {
    const session = { jsonl_usage: { deduped_message_count: 10 } };
    const map = new Map();
    const result = buildStratificationObject({
      session,
      cluster_map: map,
      session_meta_path: '/s/meta.md',
      has_facet: true,
    });
    expect(result).toMatchObject({
      write_path: 'single',
      facets_coverage_class: 'with',
      size_bucket: 'medium',
      user_message_count: 10,
    });
  });
});
