### File 1: `measurement/report/primitives.cjs`

Zero-dependency, no-color, pipe-friendly rendering helpers. Spec per RESEARCH.md §Report Layer #Primitive helpers:

```javascript
'use strict';

/**
 * Render a GitHub-flavored markdown table.
 * headers: array of column-header strings
 * rows: array of row arrays (strings or stringifiable)
 * opts.alignments: array of 'left'|'right'|'center', length matches headers (optional)
 * Empty rows render as '| (no data) |' row.
 */
function mdTable(headers, rows, opts = {}) {
  if (!Array.isArray(headers) || headers.length === 0) return '';
  const out = [];
  out.push('| ' + headers.join(' | ') + ' |');
  const sep = (headers.map((_, i) => {
    const a = (opts.alignments && opts.alignments[i]) || 'left';
    if (a === 'right') return '---:';
    if (a === 'center') return ':---:';
    return '---';
  })).join(' | ');
  out.push('| ' + sep + ' |');
  if (!Array.isArray(rows) || rows.length === 0) {
    out.push('| ' + headers.map(() => '(no data)').join(' | ') + ' |');
  } else {
    for (const row of rows) {
      const cells = headers.map((_, i) => {
        const v = row[i];
        if (v === null || v === undefined) return '—';
        // Escape pipes and newlines
        const s = String(v).replace(/\|/g, '\\|').replace(/\n/g, ' ');
        return s;
      });
      out.push('| ' + cells.join(' | ') + ' |');
    }
  }
  return out.join('\n');
}

/**
 * Render a single ASCII bar.
 * label: left side string
 * value: number (>= 0)
 * max: reference max (scales bar width)
 * width: total bar-column width in characters (default 40)
 * No color. Uses `#` for filled, space for empty.
 */
function asciiBar(label, value, max, width = 40) {
  const v = Math.max(0, Number(value) || 0);
  const m = Math.max(1, Number(max) || 1);
  const filled = Math.round((v / m) * width);
  const bar = '#'.repeat(filled) + ' '.repeat(Math.max(0, width - filled));
  return `${label} | ${bar} | ${v}`;
}

/**
 * Render a structured header block for a report.
 * Always surfaces: observed_at, extractor_registry_size, feature_row_count,
 * by_source_family missing/stale/unknown counts, overall_tier, anomaly_count, caveats.
 * This is DC-8 / G-3 — the report must expose the same epistemic provenance the JSON contract does.
 */
function headerBlock({ title, observed_at, provenance, coverage, reliability, anomaly_count, caveats }) {
  const out = [];
  out.push('```');
  out.push(`title: ${title || '(untitled)'}`);
  out.push(`observed_at: ${observed_at || 'unknown'}`);
  // provenance fields
  const extractor_size = (provenance && provenance.live_overlay && provenance.live_overlay.extractor_registry_size)
    || (provenance && provenance.extractor_registry_size)
    || 'unknown';
  const feature_row_count = (provenance && provenance.feature_row_count)
    || (provenance && provenance.live_overlay && provenance.live_overlay.feature_row_count)
    || 'unknown';
  out.push(`extractor_registry_size: ${extractor_size}`);
  out.push(`feature_row_count: ${feature_row_count}`);
  // coverage fields (by_source_family missing/stale/unknown)
  const cov = coverage || {};
  const bsf = cov.by_source_family || {};
  const missing = bsf.missing_sources || 0;
  const stale = bsf.stale_sources || 0;
  const unknown = bsf.unknown_sources || 0;
  out.push(`coverage.missing_sources: ${missing}`);
  out.push(`coverage.stale_sources: ${stale}`);
  out.push(`coverage.unknown_sources: ${unknown}`);
  // reliability
  const tier = (reliability && reliability.overall_tier) || (reliability && reliability.tier) || 'unknown';
  out.push(`reliability.overall_tier: ${tier}`);
  // anomaly
  out.push(`anomaly_count: ${typeof anomaly_count === 'number' ? anomaly_count : 'unknown'}`);
  // caveats
  if (Array.isArray(caveats) && caveats.length > 0) {
    out.push('caveats:');
    for (const c of caveats) out.push(`  - ${c}`);
  }
  out.push('```');
  return out.join('\n');
}

module.exports = { mdTable, asciiBar, headerBlock };
```

### File 2: `measurement/report/dispatch.cjs`

```javascript
'use strict';

const { queryMeasurement } = require('../query.cjs');
const { LOOP_KEYS } = require('../registry.cjs').LOOP_DEFINITIONS
  ? { LOOP_KEYS: Object.keys(require('../registry.cjs').LOOP_DEFINITIONS) }
  : require('../registry.cjs');  // fall back to direct LOOP_KEYS export if available

function renderLoopReport(loop, queryResult, opts = {}) {
  let template;
  try {
    template = require(`./loops/${loop}.cjs`);
  } catch (err) {
    throw new Error(`No report template for loop "${loop}". Known: ${LOOP_KEYS.join(', ')}`);
  }
  if (typeof template.render !== 'function') {
    throw new Error(`Report template for "${loop}" does not export render().`);
  }
  return template.render(queryResult, opts);
}

function parseReportArgs(args) {
  // args looks like: [ '<loop>', '--no-stratification', '--runtime', 'codex-cli' ]
  const opts = { stratified: true, runtime: null };
  let loop = null;
  for (let i = 0; i < args.length; i++) {
    const a = args[i];
    if (a === '--no-stratification') { opts.stratified = false; continue; }
    if (a === '--stratified') { opts.stratified = true; continue; }
    if (a === '--runtime' && args[i + 1]) { opts.runtime = args[i + 1]; i++; continue; }
    if (!loop) loop = a;
  }
  return { loop, opts };
}

function cmdMeasurementReport(cwd, args, raw) {
  const { loop, opts } = parseReportArgs(args);
  if (!loop) {
    return { error: `Usage: gsd-tools measurement report <loop>. Known loops: ${LOOP_KEYS.join(', ')}` };
  }
  if (!LOOP_KEYS.includes(loop)) {
    return { error: `Unknown loop "${loop}". Known: ${LOOP_KEYS.join(', ')}` };
  }
  // Call queryMeasurement with the loop as the question — classifyQuestion dispatches on loop keywords.
  const queryResult = queryMeasurement(cwd, { question: loop, scope: 'project', runtime: opts.runtime });
  if (raw) return queryResult;
  // Render text
  const text = renderLoopReport(loop, queryResult, opts);
  return { __text: text };  // caller prints .__text when raw=false
}

module.exports = { renderLoopReport, cmdMeasurementReport, parseReportArgs };
```

// In signal_quality.cjs render():
const facetsFeatures = queryResult.features.filter(f => f.feature && f.feature.startsWith('facets_semantic_summary'));
if (opts.stratified !== false) {

  const withoutCount = facetsFeatures.filter(f => f.availability_status === 'not_emitted').length;
