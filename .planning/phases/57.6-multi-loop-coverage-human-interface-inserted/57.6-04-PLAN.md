Spec (verbatim from RESEARCH.md §Wave 2 #1):

```javascript
const thinkingCompositeExtractor = defineExtractor({
  name: 'thinking_composite',
  source_family: 'RUNTIME',
  raw_sources: ['claude_jsonl_projects'],
  runtimes: ['claude-code'],
  reliability_tier: 'direct_observation',
  features_produced: ['thinking_composite'],
  serves_loop: ['agent_performance'],
  distinguishes: ['thinking_emission_gate', 'model_stratified_reasoning_effort'],
  status_semantics: ['exposed', 'not_emitted', 'not_applicable', 'not_available'],
  extract(extractor, context) {
    const claude = getClaudeDataset(context);
    return claude.sessions.map(session => {
      const thinking = session.thinking || { emitted: false, block_count: 0, total_chars: 0, visible_chars: 0, over_visible_ratio: null, dispatch_context: 'unavailable' };
      const model = (session.runtime_identity && session.runtime_identity.model) || '';
      // Blocker 2 fix: session.runtime_identity does NOT expose effective_effort_level in HEAD (sources/claude.cjs:818-823 is {model, claude_version, entrypoint, permission_mode}). Use session.effort_override.value directly — it IS the per-session override per sources/claude.cjs:824. If no override was observed the value is null (correct semantics — downstream reports can differentiate 'no override' from 'missing capability').
      const effort = (session.effort_override && session.effort_override.value) || null;
      // Three-level gate per spike 010 DECISION §5:
      let availability_status;
      let gate_applied;
      if (thinking.dispatch_context === 'unavailable' || session.parent_jsonl.status !== 'matched') {
        availability_status = 'not_available';
        gate_applied = null;
      } else if (/^claude-haiku/i.test(model)) {
        availability_status = 'not_applicable';
        gate_applied = 'model_family';
      } else if (thinking.dispatch_context === 'subagent') {
        availability_status = 'not_available';
        gate_applied = 'dispatch_context'; // per spike 009 — subagent dispatch emits zero blocks regardless
      } else if (thinking.block_count === 0) {
        availability_status = 'not_emitted';
        gate_applied = 'emission_threshold';
      } else {
        availability_status = 'exposed';
        gate_applied = null;
      }
      return buildFeatureRecord(extractor, {
        feature_name: `thinking_composite:${session.session_id}`,
        runtime: 'claude-code',
        availability_status,
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'direct_observation',
        value: {
          session_id: session.session_id,
          thinking_emitted: thinking.emitted,
          thinking_block_count: thinking.block_count,
          thinking_total_chars: thinking.total_chars,
          thinking_over_visible_ratio: thinking.over_visible_ratio,
          model,
          effective_effort_level: effort,
          gate_applied,
          dispatch_context: thinking.dispatch_context,
        },
        coverage: buildCoverage(session),
        provenance: { session_id: session.session_id, parent_jsonl_path: session.parent_jsonl.path || null, gate_spec: 'spike-010-DECISION.md §5' },
        freshness: combineFreshness(context.observed_at, [session.parent_jsonl.path]),
        notes: [ 'Three-level gate: model_family (Haiku→not_applicable), dispatch_context (subagent→not_available), emission_threshold (0 blocks→not_emitted).' ],
      });
    });
  },
});
```

### Extractor 2: `marker_density` (MEAS-RUNTIME-06 revised)

Add immediately after `thinking_composite`. Reuse same precondition logic (same availability gates — dense thinking required; no thinking → not_emitted). DROP branching + dead_end markers per spike 010 finding #4 (zero matches; unfit regex).

Spec:

```javascript
const MARKER_REGEX_SELF_CORRECTION = /\b(actually|wait|reconsider|correction|let me reconsider|on (second )?thought|I was wrong|revise)\b/gi;
const MARKER_REGEX_UNCERTAINTY = /\b(not sure|uncertain|might(?: not)?|unclear|perhaps|possibly|I think|may(?: not)?)\b/gi;

const markerDensityExtractor = defineExtractor({
  name: 'marker_density',
  source_family: 'RUNTIME',
  raw_sources: ['claude_jsonl_projects'],
  runtimes: ['claude-code'],
  reliability_tier: 'direct_observation',
  features_produced: ['marker_density'],
  serves_loop: ['agent_performance'],  // G-4: effort tracking, NOT quality
  distinguishes: ['effort_marker_presence', 'model_effort_stratification'],
  status_semantics: ['exposed', 'not_emitted', 'not_applicable', 'not_available'],
  extract(extractor, context) {
    const claude = getClaudeDataset(context);
    return claude.sessions.map(session => {
      // Reuse availability gate from thinking_composite — if thinking is absent, density is meaningless.
      const thinking = session.thinking || { block_count: 0, total_chars: 0, dispatch_context: 'unavailable' };
      const model = (session.runtime_identity && session.runtime_identity.model) || '';
      let availability_status;
      if (thinking.dispatch_context === 'unavailable' || session.parent_jsonl.status !== 'matched') availability_status = 'not_available';
      else if (/^claude-haiku/i.test(model)) availability_status = 'not_applicable';
      else if (thinking.dispatch_context === 'subagent') availability_status = 'not_available';
      else if (thinking.block_count === 0) availability_status = 'not_emitted';
      else availability_status = 'exposed';

      // Re-scan thinking text to count markers (session.thinking.total_chars is cached; we need the text to regex).
      // Collect thinking text per session from parent_jsonl records.
      let thinking_text = '';
      if (availability_status === 'exposed') {
        for (const rec of session.parent_jsonl.records) {
          if (rec.type !== 'assistant' || !Array.isArray(rec.message && rec.message.content)) continue;
          for (const block of rec.message.content) {
            if (block.type === 'thinking' && typeof block.text === 'string') thinking_text += block.text + '\n';
          }
        }
      }
      const self_corrections = (thinking_text.match(MARKER_REGEX_SELF_CORRECTION) || []).length;
      const uncertainties = (thinking_text.match(MARKER_REGEX_UNCERTAINTY) || []).length;
      const per_1k = (count) => thinking.total_chars > 0 ? (count * 1000) / thinking.total_chars : 0;
      return buildFeatureRecord(extractor, {
        feature_name: `marker_density:${session.session_id}`,
        runtime: 'claude-code',
        availability_status,
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'direct_observation',
        value: {
          session_id: session.session_id,
          marker_self_correction_density: per_1k(self_corrections),
          marker_uncertainty_density: per_1k(uncertainties),
          marker_self_correction_count: self_corrections,
          marker_uncertainty_count: uncertainties,
          thinking_total_chars: thinking.total_chars,
          model,
          effort_level: (session.effort_override && session.effort_override.value) || null,  // Blocker 2 fix: session.effort_override.value is the authoritative per-session override; runtime_identity has no effective_effort_level in HEAD
          caveat: 'effort_tracking_not_quality_proxy',  // G-4: literal caveat rendered in report
          dropped_markers: ['branching', 'dead_end'],   // spike 010 #4: zero-match regex unfit for Claude vocabulary
        },
        coverage: buildCoverage(session),
        provenance: { session_id: session.session_id, regex_spec: 'spike-010-DECISION.md §"Summary complexity extractor UPDATED SHAPE" + finding #5' },
        freshness: combineFreshness(context.observed_at, [session.parent_jsonl.path]),
        notes: [
          'G-4: Marker density is effort-tracking, NEVER reasoning quality. Spike 010 qualitative_comparison.md finding #6 grounds this.',
          'Branching + dead_end markers DROPPED per spike 010 finding #4 (zero matches across 21 thinking-emissions).',
        ],
      });
    });
  },
});
```
