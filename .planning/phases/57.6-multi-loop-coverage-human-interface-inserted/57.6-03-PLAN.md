### Helper 1: `scanThinkingBlocks(records)` â€” internal, called from loadClaude

```javascript
// Adjacent to aggregateAssistantUsage/extractEffortOverride (around line 300-400).
// Gate #1 per spike 010 DECISION.md: model family (Haiku -> not_applicable) is applied by the EXTRACTOR,
// not here. This scanner only counts blocks.
// Gate #2 per spike 009: isSidechain distinguishes subagent from parent dispatch; returned as dispatch_context.
// Gate #3 per spike 010: emission threshold (capable but zero blocks) is detected as block_count === 0
// and surfaced by the extractor as not_emitted.

function scanThinkingBlocks(records) {
  if (!Array.isArray(records) || records.length === 0) {
    return { emitted: false, block_count: 0, total_chars: 0, visible_chars: 0, over_visible_ratio: null, dispatch_context: 'parent' };
  }
  let block_count = 0;
  let total_chars = 0;
  let visible_chars = 0;
  // First-record sidechain detection (per spike 009: only the first assistant record carries isSidechain on the session-level dispatch context)
  let dispatch_context = 'parent';
  for (const record of records) {
    if (record && record.isSidechain === true) {
      dispatch_context = 'subagent';
      break;
    }
  }
  for (const record of records) {
    if (!record || record.type !== 'assistant') continue;
    const content = record.message && record.message.content;
    if (!Array.isArray(content)) continue;
    for (const block of content) {
      if (!block || typeof block !== 'object') continue;
      if (block.type === 'thinking' && typeof block.text === 'string') {
        block_count++;
        total_chars += block.text.length;
      } else if (block.type === 'text' && typeof block.text === 'string') {
        visible_chars += block.text.length;
      }
    }
  }
  const over_visible_ratio = visible_chars > 0 ? total_chars / visible_chars : null;
  return {
    emitted: block_count > 0,
    block_count,
    total_chars,
    visible_chars,
    over_visible_ratio,
    dispatch_context,
  };
}
```

### Helper 2: `countClearInvocations(records)` â€” exported for extractor use

```javascript
// Scans user records for <command-name>/clear</command-name> markers.
// Reuses HUMAN_TURN_COMMAND_PREFIXES parsing pattern (sources/claude.cjs:22-27).
function countClearInvocations(records) {
  if (!Array.isArray(records)) return 0;
  let count = 0;
  for (const record of records) {
    if (!record || record.type !== 'user') continue;
    const content = record.message && record.message.content;
    let text = '';
    if (typeof content === 'string') text = content;
    else if (Array.isArray(content) && content[0] && typeof content[0].text === 'string') text = content[0].text;
    if (text.includes('<command-name>/clear</command-name>')) count++;
  }
  return count;
}
```

### Helper 3: `extractCompactionEvents(records)` â€” exported for extractor use

```javascript
// Scans for compaction markers per anomaly-stress-tests E6.2:
// - system records with subtype === 'compact_boundary' (compact_metadata.trigger âˆˆ {manual, auto}, compact_metadata.pre_tokens)
// - assistant records with isCompactSummary === true
// Returns { count, triggers: {manual, auto}, pre_tokens: [...], summary_messages: N }.
// Current 211-session corpus yields zero per E6.2 Layer 1 â€” this is DATA, not a bug (Pitfall 5).
function extractCompactionEvents(records) {
  const result = { count: 0, triggers: { manual: 0, auto: 0 }, pre_tokens: [], summary_messages: 0 };
  if (!Array.isArray(records)) return result;
  for (const record of records) {
    if (!record) continue;
    if (record.type === 'system' && record.subtype === 'compact_boundary') {
      result.count++;
      const meta = record.compact_metadata || {};
      if (meta.trigger === 'manual') result.triggers.manual++;
      else if (meta.trigger === 'auto') result.triggers.auto++;
      if (typeof meta.pre_tokens === 'number') result.pre_tokens.push(meta.pre_tokens);
    }
    if (record.type === 'assistant' && record.isCompactSummary === true) {
      result.summary_messages++;
    }
  }
  return result;
}
```

```javascript
const fs = require('fs');
// Note: fs likely already required at top of file; confirm before adding duplicate require.

/**
 * Scan a Codex rollout JSONL file for compaction events.
 * Per anomaly-stress-tests E6.2 + REQUIREMENTS MEAS-RUNTIME-07:
 * - event_msg records with type === 'context_compacted' â†’ each is one compaction event
 * - Each may carry a replacement_history array; capture its length.
 *
 * Returns { count: N, replacement_history_lengths: [...], context_compacted_events: N, scanned: bool, error: string|null }.
 *
 * Error handling:
 * - Missing file â†’ { ..., count: 0, scanned: false, error: 'file_missing' }
 * - Parse error (malformed JSONL line) â†’ skip line, continue; set error to 'partial_parse' if any line failed.
 * - Empty file â†’ { ..., count: 0, scanned: true, error: null }
 */
function scanRolloutEvents(rollout_path) {
  const result = {
    count: 0,
    replacement_history_lengths: [],
    context_compacted_events: 0,
    scanned: false,
    error: null,
  };
  if (!rollout_path || typeof rollout_path !== 'string') {
    result.error = 'no_path';
    return result;
  }
  if (!fs.existsSync(rollout_path)) {
    result.error = 'file_missing';
    return result;
  }
  let raw;
  try {
    raw = fs.readFileSync(rollout_path, 'utf8');
  } catch (err) {
    result.error = `read_error: ${err.code || err.message}`;
    return result;
  }
  result.scanned = true;
  const lines = raw.split(/\r?\n/);
  let partial = false;
  for (const line of lines) {
    if (!line.trim()) continue;
    let obj;
    try {
      obj = JSON.parse(line);
    } catch (_e) {
      partial = true;
      continue;
    }
    // Codex rollout event_msg shape: { type: 'event_msg', payload: { type: 'context_compacted', replacement_history: [...] } }
    // OR flat: { type: 'context_compacted', replacement_history: [...] }
    // Defensive: check both shapes.
    const payload = obj && (obj.payload || obj);
    if (payload && payload.type === 'context_compacted') {
      result.count++;
      result.context_compacted_events++;
      if (Array.isArray(payload.replacement_history)) {
        result.replacement_history_lengths.push(payload.replacement_history.length);
      }
    }
  }
  if (partial) result.error = 'partial_parse';
  return result;
}
```
