```javascript
const facetsSemanticSummaryExtractor = defineExtractor({
  name: 'facets_semantic_summary',
  source_family: 'DERIVED',
  raw_sources: ['claude_facets', 'claude_session_meta'],
  runtimes: ['claude-code'],
  reliability_tier: 'artifact_derived',  // LLM-extracted, not direct observation
  features_produced: ['facets_semantic_summary'],
  serves_loop: ['signal_quality', 'agent_performance'],
  distinguishes: ['semantic_goal_distribution', 'friction_surface_map'],
  status_semantics: ['derived', 'not_available', 'not_emitted'],
  extract(extractor, context) {
    const claude = getClaudeDataset(context);
    // Build cluster map ONCE for the entire session corpus (shared with write_path extractor).
    // Collect all session-meta + facet file mtimes.
    const allFiles = [];
    for (const session of claude.sessions) {
      if (session.session_meta && session.session_meta.path && session.session_meta.mtime_ms) {
        allFiles.push({ path: session.session_meta.path, mtime: session.session_meta.mtime_ms });
      }
      if (session.facets && session.facets.path && session.facets.mtime_ms) {
        allFiles.push({ path: session.facets.path, mtime: session.facets.mtime_ms });

    const cluster_map = clusterByMtime(allFiles);
    return claude.sessions.map(session => {
      const has_facet = !!(session.facets && session.facets.record);
      const facet = has_facet ? session.facets.record : null;
      const stratification = buildStratificationObject({
        session,
        cluster_map,
        session_meta_path: session.session_meta && session.session_meta.path,

      });
      // DC-4 enforcement: stratification always present on every row.

          reliability_tier: 'artifact_derived',
          value: { session_id: session.session_id, stratification },
          coverage: buildCoverage(session),
          provenance: { session_id: session.session_id, facet_path: null },
          freshness: combineFreshness(context.observed_at, []),
          notes: ['No facet file matched for this session — facets coverage asymmetry per E5.8 Finding C; absence is data.'],
        });
      }
      return buildFeatureRecord(extractor, {
        feature_name: `facets_semantic_summary:${session.session_id}`,
        runtime: 'claude-code',
        availability_status: 'derived',
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'artifact_derived',

          friction_counts: facet.friction_counts || null,
          friction_detail: facet.friction_detail || null,
          primary_success: facet.primary_success || null,
          brief_summary: facet.brief_summary || null,
          stratification,  // DC-4: MANDATORY
```

```javascript
const derivedWritePathProvenanceExtractor = defineExtractor({
  name: 'derived_write_path_provenance',
  source_family: 'DERIVED',
  raw_sources: ['claude_session_meta', 'claude_facets'],
  runtimes: ['claude-code'],
  reliability_tier: 'artifact_derived',
  features_produced: ['derived_write_path_provenance'],
  serves_loop: ['signal_quality', 'pipeline_integrity'],
  distinguishes: ['bulk_vs_single_write_path', 'mtime_cluster_membership'],
  status_semantics: ['derived', 'not_available'],
  extract(extractor, context) {
    const claude = getClaudeDataset(context);
    // Build same cluster map (or memoise via context if feasible; for now build per extractor — cheap O(N log N)).
    const allFiles = [];
    for (const session of claude.sessions) {
      if (session.session_meta && session.session_meta.path && session.session_meta.mtime_ms) {
        allFiles.push({ path: session.session_meta.path, mtime: session.session_meta.mtime_ms, kind: 'session_meta' });
      }
      if (session.facets && session.facets.path && session.facets.mtime_ms) {
        allFiles.push({ path: session.facets.path, mtime: session.facets.mtime_ms, kind: 'facet' });
      }
    }
    const cluster_map = clusterByMtime(allFiles);
    // Emit one row per artifact (session_meta OR facet), not per session.
    const rows = [];
    for (const file of allFiles) {
      // Find cluster membership
      let cluster_id = null;
      let cluster_size = null;
      let window_seconds = null;
      let write_path = 'single';
      for (const [cid, c] of cluster_map.entries()) {
        if (c.file_set.has(file.path)) {
          cluster_id = cid;
          cluster_size = c.size;
          window_seconds = c.window_seconds;
          write_path = 'bulk';
          break;
        }
      }
      rows.push(buildFeatureRecord(extractor, {
        feature_name: `derived_write_path_provenance:${file.path}`,
        runtime: 'claude-code',
        availability_status: 'derived',
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'artifact_derived',
        value: {
          artifact_path: file.path,
          artifact_kind: file.kind,
          modified_at: new Date(file.mtime).toISOString(),
          mtime_cluster_id: cluster_id,
          write_path,
          cluster_size,
          cluster_window_seconds: window_seconds,
        },
        coverage: { artifact_path: file.path, artifact_kind: file.kind },
        provenance: { cluster_algorithm: 'stratify.cjs:clusterByMtime window=2s min_cluster=5 per E5.8 Finding A' },
        freshness: combineFreshness(context.observed_at, [file.path]),
        notes: ['Classification is snapshot at observed_at. Re-running rebuild during an /insights run may shift membership — that is data per G-6, not a defect.'],
      }));
    }
    return rows;
  },
});
```

### Extractor 3: `insights_mass_rewrite_boundary` (MEAS-DERIVED-04)

Spec (verbatim from RESEARCH.md §Wave 2 #8):

```javascript
const insightsMassRewriteBoundaryExtractor = defineExtractor({
  name: 'insights_mass_rewrite_boundary',
  source_family: 'DERIVED',
  raw_sources: ['claude_session_meta', 'claude_facets'],
  runtimes: ['claude-code'],
  reliability_tier: 'artifact_derived',
  features_produced: ['insights_mass_rewrite_boundary'],
  serves_loop: ['signal_quality', 'cross_session_patterns'],
  distinguishes: ['insights_generation_batch', 'analysis_staleness_detection'],
  status_semantics: ['derived', 'not_available'],
  extract(extractor, context) {
    const claude = getClaudeDataset(context);
    const sessionFiles = [];
    for (const session of claude.sessions) {
      if (session.session_meta && session.session_meta.path && session.session_meta.mtime_ms) {
        sessionFiles.push({ path: session.session_meta.path, mtime: session.session_meta.mtime_ms, session_id: session.session_id, session });
      }
    }
    const cluster_map = clusterByMtime(sessionFiles);
    // One row per detected /insights batch
    const rows = [];
    for (const [batch_id, cluster] of cluster_map.entries()) {
      // Get session_ids for files in this cluster
      const filesInCluster = sessionFiles.filter(sf => cluster.file_set.has(sf.path));
      const session_ids = filesInCluster.map(sf => sf.session_id);
      // Staleness detection: compare session_meta mtime vs parent JSONL mtime per session
      const sessions_with_newer_jsonl = [];
      for (const sf of filesInCluster) {
        const jsonl_mtime = sf.session.parent_jsonl && sf.session.parent_jsonl.mtime_ms;
        if (typeof jsonl_mtime === 'number' && jsonl_mtime > sf.mtime) {
          sessions_with_newer_jsonl.push(sf.session_id);
        }
      }
      rows.push(buildFeatureRecord(extractor, {
        feature_name: `insights_mass_rewrite_boundary:${batch_id}`,
        runtime: 'claude-code',
        availability_status: 'derived',
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'artifact_derived',
        value: {
          batch_id,
          batch_mtime_window: { start: cluster.window_start, end: cluster.window_end, duration_seconds: cluster.window_seconds },
          session_ids_in_batch: session_ids,
          batch_size: cluster.size,
          staleness: {
            sessions_with_newer_jsonl,
            stale_analysis_detected: sessions_with_newer_jsonl.length > 0,
          },
        },
        coverage: { batch_id, session_count: cluster.size },
        provenance: { cluster_algorithm: 'stratify.cjs:clusterByMtime window=2s min_cluster=5' },
        freshness: combineFreshness(context.observed_at, filesInCluster.map(sf => sf.path)),
        notes: ['A /insights batch is detected when ≥5 session-meta files share mtime within 2s. Staleness: JSONL appended after last batch.'],
      }));
    }
    return rows;
  },
});
```

```javascript
const fs = require('fs');
const path = require('path');

const skipReasonCanonicalExtractor = defineExtractor({
  name: 'skip_reason_canonical',
  source_family: 'GSDR',
  raw_sources: ['planning_config'],
  runtimes: ['project'],
  reliability_tier: 'direct_observation',
  features_produced: ['skip_reason_canonical'],
  serves_loop: ['signal_quality', 'pipeline_integrity'],
  distinguishes: ['skip_reason_vocabulary_drift', 'non_canonical_skip_reasons'],
  status_semantics: ['exposed', 'not_available', 'not_emitted'],
  extract(extractor, context) {
    const gsdr = context.gsdrRaw || {};
    const config = gsdr.planning_config || null;
    if (!config) {
      return [buildFeatureRecord(extractor, {
        feature_name: 'skip_reason_canonical',
        runtime: 'project',
        availability_status: 'not_available',
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'direct_observation',
        value: { reason: 'no .planning/config.json' },
        coverage: {},
        provenance: {},
        freshness: combineFreshness(context.observed_at, []),
        notes: ['Config absent; extractor cannot evaluate skip_reason canonicity.'],
      })];
    }
    // Load canonical enum from feature-manifest.json
    const manifestPath = path.resolve(__dirname, '..', '..', '..', 'feature-manifest.json');
    let canonical = [];
    try {
      canonical = JSON.parse(fs.readFileSync(manifestPath, 'utf8')).automation_skip_reasons || [];
    } catch (err) {
      canonical = [];  // fall through: extractor will emit not_available below
    }
    if (canonical.length === 0) {
      return [buildFeatureRecord(extractor, {
        feature_name: 'skip_reason_canonical',
        runtime: 'project',
        availability_status: 'not_available',
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'direct_observation',
        value: { reason: 'feature-manifest.automation_skip_reasons missing or empty' },
        coverage: {},
        provenance: { manifest_path: manifestPath },
        freshness: combineFreshness(context.observed_at, [manifestPath]),
        notes: ['Canonical enum unavailable — cannot classify.'],
      })];
    }
    const stats = (config.automation && config.automation.stats) || {};
    const featureKeys = Object.keys(stats);
    if (featureKeys.length === 0) {
      return [buildFeatureRecord(extractor, {
        feature_name: 'skip_reason_canonical',
        runtime: 'project',
        availability_status: 'not_emitted',
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'direct_observation',
        value: { canonical_enum: canonical, count_by_reason: {} },
        coverage: { feature_count: 0 },
        provenance: { manifest_path: manifestPath },
        freshness: combineFreshness(context.observed_at, [manifestPath]),
        notes: ['No automation features tracked yet.'],
      })];
    }
    return featureKeys.map(feature_key => {
      const s = stats[feature_key] || {};
      const last_skip_reason = s.last_skip_reason || null;
      const canonical_match = last_skip_reason && canonical.includes(last_skip_reason);
      return buildFeatureRecord(extractor, {
        feature_name: `skip_reason_canonical:${feature_key}`,
        runtime: 'project',
        availability_status: last_skip_reason ? 'exposed' : 'not_emitted',
        symmetry_marker: 'asymmetric_only',
        reliability_tier: 'direct_observation',
        value: {
          feature_key,
          last_skip_reason,
          canonical: !!canonical_match,
          canonical_match: canonical_match ? last_skip_reason : 'unknown',
          count_by_reason: null,  // aggregation across time series would require history; not stored in config.json
        },
        coverage: { feature_key },
        provenance: { manifest_path: manifestPath, config_key: `automation.stats.${feature_key}.last_skip_reason` },
        freshness: combineFreshness(context.observed_at, [manifestPath]),
        notes: canonical_match ? [] : ['Non-canonical skip_reason observed; may indicate vocabulary drift.'],
      });
    });
  },
});
```
