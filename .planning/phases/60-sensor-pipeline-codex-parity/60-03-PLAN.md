```python
#!/usr/bin/env python3
"""
extract-session-fingerprints.py — Phase 60 cross-runtime log-sensor extractor.

Usage: python3 extract-session-fingerprints.py <session-jsonl-path>
Output: Single JSON line on stdout with the normalized fingerprint schema.
Errors: Parse/format failures are embedded in the output as SENS-07 diagnostic fields,
        NOT raised as exceptions (DC-2: sensor emits candidates; synthesizer arbitrates).

Dependencies: Python 3 stdlib only (json, sys, collections).
"""
import json
import sys
from collections import Counter


# ---- Known-event vocabularies (2026-04-21 live-corpus audit) ----
CODEX_KNOWN_EVENT_MSG_TYPES = {
    # Conversational / fingerprint counters
    'user_message', 'agent_message',
    # Token accounting
    'token_count',
    # Interruptions
    'turn_aborted',
    # Known-but-no-fingerprint
    'task_started', 'task_complete',
    'exec_command_end', 'patch_apply_end',
    'context_compacted',
    'collab_waiting_end', 'collab_agent_spawn_end',
    'collab_close_end', 'collab_agent_interaction_end',
    'mcp_tool_call_end',
    'item_completed', 'thread_rolled_back',
    'web_search_end',
}
CODEX_KNOWN_RESPONSE_ITEM_TYPES = {
    'function_call', 'function_call_output',
    'reasoning', 'message',
    'custom_tool_call', 'custom_tool_call_output',
    'web_search_call',
}


def detect_format(path):
    """Return 'claude' | 'codex' | None. G-5: detects LOG FILE runtime, not harness runtime."""
    try:
        with open(path, 'r') as fp:
            first = fp.readline()
        obj = json.loads(first)
    except (IOError, json.JSONDecodeError):
        return None
    t = obj.get('type')
    if t == 'session_meta':
        return 'codex'
    if t in ('system', 'user', 'assistant', 'summary'):
        return 'claude'
    return None


def _empty_fingerprint():
    return {
        'session_id': None,
        'start_time': None,
        'end_time': None,
        'user_message_count': 0,
        'assistant_message_count': 0,
        'tool_call_count': 0,
        'tool_error_count': 0,
        'total_tokens': 0,
        'model': None,
        'interruptions': 0,
        'direction_changes': None,  # Stage 3c text-pattern pass fills this
        # Codex-only additive fields (G-2: never drop; 'not_available' on Claude)
        'reasoning_output_tokens': 'not_available',
        'rate_limit_primary_used_percent': 'not_available',
        'model_context_window': 'not_available',
        'source': 'not_available',  # Codex session_meta.payload.source: 'exec' | 'cli'
        'agent_role': 'not_available',  # Codex subagent marker (Q1 closure)
        # SENS-07 diagnostics (arrays, emitted per file)
        '_sens07_parse_errors': [],          # list of {line_number, snippet}
        '_sens07_unknown_event_msg_types': {},   # counter: {type_name: count}
        '_sens07_unknown_response_item_types': {},
    }


def extract_claude_fingerprint(path):
    """Parse Claude Code JSONL session log. Matches existing gsd-log-sensor.md Stage 1c logic."""
    fp_result = _empty_fingerprint()
    fp_result['_format'] = 'claude'
    parse_errors = []
    try:
        with open(path, 'r') as fp:
            for lineno, line in enumerate(fp, 1):
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                except json.JSONDecodeError as e:
                    parse_errors.append({'line_number': lineno, 'snippet': line[:120]})
                    continue
                t = obj.get('type')
                ts = obj.get('timestamp')
                if ts and not fp_result['start_time']:
                    fp_result['start_time'] = ts
                if ts:
                    fp_result['end_time'] = ts
                if t == 'user':
                    fp_result['user_message_count'] += 1
                elif t == 'assistant':
                    fp_result['assistant_message_count'] += 1
                    msg = obj.get('message', {})
                    if isinstance(msg, dict):
                        usage = msg.get('usage', {}) or {}
                        fp_result['total_tokens'] = (usage.get('input_tokens') or 0) + (usage.get('output_tokens') or 0) + fp_result['total_tokens']
                        fp_result['model'] = msg.get('model', fp_result['model'])
                        content = msg.get('content', [])
                        if isinstance(content, list):
                            for item in content:
                                if isinstance(item, dict) and item.get('type') == 'tool_use':
                                    fp_result['tool_call_count'] += 1
                elif t == 'summary':
                    # Summary events — Claude's conversation compaction marker
                    pass
    except IOError as e:
        parse_errors.append({'line_number': 0, 'snippet': f'IOError: {e}'})
    fp_result['_sens07_parse_errors'] = parse_errors
    return fp_result


def extract_codex_fingerprint(path):
    """Parse Codex CLI rollout JSONL. New logic; normalizes to same schema."""
    fp_result = _empty_fingerprint()
    fp_result['_format'] = 'codex'
    parse_errors = []
    unknown_msg = Counter()
    unknown_resp = Counter()
    try:
        with open(path, 'r') as fp:
            for lineno, line in enumerate(fp, 1):
                line = line.strip()
                if not line:
                    continue
                try:
                    obj = json.loads(line)
                except json.JSONDecodeError:
                    parse_errors.append({'line_number': lineno, 'snippet': line[:120]})
                    continue
                t = obj.get('type')
                payload = obj.get('payload', {}) if isinstance(obj.get('payload'), dict) else {}
                ts = obj.get('timestamp')
                if ts and not fp_result['start_time']:
                    fp_result['start_time'] = ts
                if ts:
                    fp_result['end_time'] = ts
                if t == 'session_meta':
                    fp_result['session_id'] = payload.get('id')
                    fp_result['source'] = payload.get('source') or 'not_available'
                    # Q1 closure: agent_role marks subagent rollouts
                    agent_role = payload.get('agent_role')
                    if agent_role:
                        fp_result['agent_role'] = agent_role
                elif t == 'turn_context':
                    if not fp_result['model']:
                        fp_result['model'] = payload.get('model')
                elif t == 'event_msg':
                    pt = payload.get('type')
                    if pt == 'user_message':
                        fp_result['user_message_count'] += 1
                    elif pt == 'agent_message':
                        fp_result['assistant_message_count'] += 1
                    elif pt == 'token_count':
                        info = payload.get('info', {}) or {}
                        ttu = info.get('total_token_usage', {}) or {}
                        fp_result['total_tokens'] = ttu.get('total_tokens', fp_result['total_tokens'])
                        reasoning = ttu.get('reasoning_output_tokens')
                        if reasoning is not None:
                            fp_result['reasoning_output_tokens'] = reasoning
                        mcw = ttu.get('model_context_window')
                        if mcw is not None:
                            fp_result['model_context_window'] = mcw
                        rate_limits = payload.get('rate_limits', {}) or {}
                        primary = rate_limits.get('primary', {}) or {}
                        rlp = primary.get('used_percent')
                        if rlp is not None:
                            fp_result['rate_limit_primary_used_percent'] = rlp
                    elif pt == 'turn_aborted':
                        fp_result['interruptions'] += 1
                    elif pt in CODEX_KNOWN_EVENT_MSG_TYPES:
                        pass  # known-but-no-counter; counted/ignored
                    else:
                        unknown_msg[pt or '<null>'] += 1
                elif t == 'response_item':
                    pt = payload.get('type')
                    if pt == 'function_call':
                        fp_result['tool_call_count'] += 1
                    elif pt == 'function_call_output':
                        output = payload.get('output', '')
                        success = payload.get('success')
                        if success is False or 'error' in str(output).lower():
                            fp_result['tool_error_count'] += 1
                    elif pt in CODEX_KNOWN_RESPONSE_ITEM_TYPES:
                        pass
                    else:
                        unknown_resp[pt or '<null>'] += 1
                elif t == 'compacted':
                    pass  # valid but no fingerprint counter
                else:
                    # Unknown top-level type
                    unknown_msg[f'<top:{t}>'] += 1
    except IOError as e:
        parse_errors.append({'line_number': 0, 'snippet': f'IOError: {e}'})
    fp_result['_sens07_parse_errors'] = parse_errors
    fp_result['_sens07_unknown_event_msg_types'] = dict(unknown_msg)
    fp_result['_sens07_unknown_response_item_types'] = dict(unknown_resp)
    return fp_result


def main():
    if len(sys.argv) != 2:
        print(json.dumps({'_format': 'error', '_sens07_error': 'missing_path_arg'}))
        sys.exit(0)  # DC-2: never crash the sensor harness
    path = sys.argv[1]
    fmt = detect_format(path)
    if fmt == 'claude':
        fp_result = extract_claude_fingerprint(path)
    elif fmt == 'codex':
        fp_result = extract_codex_fingerprint(path)
    else:
        fp_result = {
            '_format': 'unknown',
            '_sens07_error': 'format_detection_failed',
            '_path': path,
            '_sens07_parse_errors': [],
            '_sens07_unknown_event_msg_types': {},
            '_sens07_unknown_response_item_types': {},
        }
    print(json.dumps(fp_result))


if __name__ == '__main__':
    main()
```
