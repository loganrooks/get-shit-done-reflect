/**
 * Phase 60 / SENS-06: Post-install cross-runtime parity check.
 *
 * Runs after reportLocalPatches() on both Claude and Codex branches.
 * Detects whether the OTHER runtime has a GSD install, compares versions,
 * writes gsd-parity-report.json artifact, prints advisory stdout line.
 *
 * Advisory-only: never fails install, never prompts (CI-safe), never auto-triggers other-runtime install.
 * G-4: honest skip when other runtime not installed or manifest unreadable.
 *
 * @param {string} targetDir - Where THIS runtime was just installed (e.g., /home/user/.claude)
 * @param {'claude'|'codex'} thisRuntime
 * @param {boolean} isGlobal - Whether THIS install is global (~/.claude) or project-local (./.claude)
 * @returns {object} Report object (also written to disk as gsd-parity-report.json)
 */
function checkCrossRuntimeParity(targetDir, thisRuntime, isGlobal) {
  const otherRuntime = thisRuntime === 'claude' ? 'codex' : 'claude';
  const otherDirName = getDirName(otherRuntime);   // '.claude' or '.codex'
  // Compute other-runtime VERSION file path using the same precedence as Phase 58.1 DC-3
  const otherScopeVersionPath = isGlobal
    ? path.join(getGlobalDir(otherRuntime, explicitConfigDir), 'get-shit-done-reflect', 'VERSION')
    : path.join(process.cwd(), otherDirName, 'get-shit-done-reflect', 'VERSION');

  const checked_at = new Date().toISOString();

  // G-4 honest skip: other runtime not installed
  if (!fs.existsSync(otherScopeVersionPath)) {
    const report = {
      schema_version: 1,
      this_runtime: thisRuntime,
      this_version: null,  // filled below
      other_runtime: otherRuntime,
      other_version: null,
      divergent: false,
      reason: 'other_runtime_not_installed',
      checked_at,
      remediation_command: null,
    };
    try {
      const thisManifestPath = path.join(targetDir, MANIFEST_NAME);
      if (fs.existsSync(thisManifestPath)) {
        report.this_version = JSON.parse(fs.readFileSync(thisManifestPath, 'utf8')).version || null;
      }
      fs.writeFileSync(path.join(targetDir, 'gsd-parity-report.json'), JSON.stringify(report, null, 2));
    } catch { /* writing the report is best-effort; do not fail install */ }
    return report;
  }

  // Read other runtime version
  let otherVersion = null;
  let otherManifest = null;
  const otherInstallDir = path.dirname(path.dirname(otherScopeVersionPath));  // strip /get-shit-done-reflect/VERSION
  try {
    otherVersion = fs.readFileSync(otherScopeVersionPath, 'utf8').trim();
    const otherManifestPath = path.join(otherInstallDir, MANIFEST_NAME);
    if (fs.existsSync(otherManifestPath)) {
      otherManifest = JSON.parse(fs.readFileSync(otherManifestPath, 'utf8'));
    }
  } catch {
    // G-4: unreadable — honest skip
    const report = {
      schema_version: 1,
      this_runtime: thisRuntime,
      this_version: null,
      other_runtime: otherRuntime,
      other_version: null,
      divergent: false,
      reason: 'other_manifest_unreadable',
      checked_at,
      remediation_command: null,
    };
    try {
      const thisManifestPath = path.join(targetDir, MANIFEST_NAME);
      if (fs.existsSync(thisManifestPath)) {
        report.this_version = JSON.parse(fs.readFileSync(thisManifestPath, 'utf8')).version || null;
      }
      fs.writeFileSync(path.join(targetDir, 'gsd-parity-report.json'), JSON.stringify(report, null, 2));
    } catch { /* best-effort */ }
    return report;
  }

  // Read this version
  let thisVersion = null;
  try {
    const thisManifestPath = path.join(targetDir, MANIFEST_NAME);
    if (fs.existsSync(thisManifestPath)) {
      thisVersion = JSON.parse(fs.readFileSync(thisManifestPath, 'utf8')).version || null;
    }
  } catch { /* thisVersion stays null */ }

  // Compare (strip +dev suffix for semantic equality; e.g., 1.19.8 == 1.19.8+dev)
  const normalize = v => (v || '').replace(/\+dev$/, '').trim();
  const divergent = thisVersion != null && otherVersion != null && normalize(otherVersion) !== normalize(thisVersion);

  const report = {
    schema_version: 1,
    this_runtime: thisRuntime,
    this_version: thisVersion,
    other_runtime: otherRuntime,
    other_version: otherVersion,
    divergent,
    divergent_file_count: 0,
    checked_at,
    remediation_command: divergent ? `node bin/install.js --${otherRuntime}` : null,
  };

  // Count divergent files when both manifests exist
  if (otherManifest && divergent) {
    try {
      const thisManifest = JSON.parse(fs.readFileSync(path.join(targetDir, MANIFEST_NAME), 'utf8'));
      const thisFiles = Object.keys(thisManifest.files || {});
      const otherFiles = Object.keys(otherManifest.files || {});
      const thisOnly = thisFiles.filter(f => !otherFiles.includes(f)).length;
      const otherOnly = otherFiles.filter(f => !thisFiles.includes(f)).length;
      report.divergent_file_count = thisOnly + otherOnly;
    } catch { /* keep 0 */ }
  }

  // Write JSON artifact
  try {
    fs.writeFileSync(path.join(targetDir, 'gsd-parity-report.json'), JSON.stringify(report, null, 2));
  } catch { /* best-effort */ }

  // Advisory stdout (non-interactive, CI-safe)
  // Use ANSI color constants (`yellow`, `cyan`, `reset`) from the outer scope of bin/install.js
  // so the parity advisory is visually consistent with `saveLocalPatches`/`reportLocalPatches` output.
  if (divergent) {
    console.log('');
    console.log(`  ${yellow}Cross-runtime parity:${reset} ${otherRuntime} is at v${otherVersion}, this install is at v${thisVersion}.`);
    if (report.divergent_file_count > 0) {
      console.log(`  ${report.divergent_file_count} file(s) differ.`);
    }
    console.log(`  To sync: ${cyan}${report.remediation_command}${reset}`);
    console.log('');
  }

  return report;
}
