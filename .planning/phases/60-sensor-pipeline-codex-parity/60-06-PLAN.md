```javascript
// get-shit-done/bin/lib/xrt02-validator.cjs
// Phase 60 / XRT-02: Patch-compatibility validator.
// Composes over bin/install.js primitives (Q4 closure: live call, not static snapshot).
// Shares classification vocabulary with patch-classifier.cjs (Plan 60-04).

const path = require('node:path');
const fs = require('node:fs');

// Live call per Q4 closure (DC-6: don't hand-roll)
const {
  claudeToCodexTools,
  replacePathsInContent,
  convertClaudeToCodexMarkdown,
  convertClaudeToCodexAgentToml,
  extractFrontmatterAndBody,
  extractFrontmatterField,
} = require('../../../bin/install.js');

// Share vocabulary with patch-classifier
const { artifactCategoryApplies, loadCapabilityMatrix } = require('./patch-classifier.cjs');

/**
 * Detect target-runtime format based on relPath and file extension.
 * Returns 'markdown' | 'toml' | 'unknown'.
 */
function detectFormat(relPath) {
  if (relPath.endsWith('.md') || relPath.endsWith('.SKILL.md')) return 'markdown';
  if (relPath.endsWith('.toml')) return 'toml';
  return 'unknown';
}

/**
 * Runtime-axis check: does this patch reference runtime-specific surfaces
 * (hooks, tools) that do not exist on the target runtime?
 */
function runtimeAxis({ patchContent, targetRuntime, relPath, capabilityMatrixText }) {
  // Representability check via patch-classifier helper
  const repr = artifactCategoryApplies(relPath, targetRuntime, capabilityMatrixText);
  if (!repr.applies) {
    return { compatible: false, class: 'feature-gap', reason: repr.reason };
  }
  // Tool references check for Claude → Codex direction
  if (targetRuntime === 'codex') {
    // Find Claude tool names referenced in the patch
    const claudeToolNames = Object.keys(claudeToCodexTools || {});
    const referencedClaudeTools = claudeToolNames.filter(tool => {
      // Match the tool as a bare identifier (whole-word)
      const re = new RegExp(`\\b${tool}\\b`);
      return re.test(patchContent);
    });
    if (referencedClaudeTools.length > 0) {
      // All found Claude tools have Codex equivalents in the map → convertible, not a feature-gap
      const unmappable = referencedClaudeTools.filter(t => !claudeToCodexTools[t]);
      if (unmappable.length > 0) {
        return {
          compatible: false,
          class: 'feature-gap',
          reason: `claude-tools-with-no-codex-mapping: ${unmappable.join(', ')}`,
        };
      }
      // Else: convertible — fall through to format-axis / conversion-axis
    }
  }
  return { compatible: true };
}

/**
 * Format-axis check: is the patch file format supported on target runtime?
 * Markdown agents run on both. TOML agents are Codex-native. Skills (SKILL.md) are Codex.
 * Frontmatter shape check: Codex agent TOML uses different field names than Claude agent YAML.
 */
function formatAxis({ patchContent, relPath, targetRuntime, patchSourceRuntime }) {
  const fmt = detectFormat(relPath);
  if (fmt === 'unknown') {
    return { compatible: true, low_confidence: true, reason: 'unknown_format_assumed_text' };
  }
  // Agent markdown with YAML frontmatter — universal
  // Agent TOML — Codex only; if target is Claude, feature-gap
  if (fmt === 'toml' && targetRuntime === 'claude') {
    return { compatible: false, class: 'feature-gap', reason: 'toml-format-is-codex-only' };
  }
  // SKILL.md — Codex-specific skill format
  if (relPath.includes('SKILL.md') && targetRuntime === 'claude') {
    return { compatible: false, class: 'feature-gap', reason: 'skill-md-is-codex-skills-surface' };
  }
  return { compatible: true };
}

/**
 * Version-axis check: does the patch carry a version marker incompatible with the target install?
 * Patches from v1.17.5 may reference file paths that have been modularized away (see golden fixture).
 */
function versionAxis({ patchContent, patchSourceVersion, targetVersion }) {
  // If we have explicit version metadata, check major version compatibility
  if (patchSourceVersion && targetVersion) {
    const srcMajor = parseInt(String(patchSourceVersion).split('.')[0] || '0', 10);
    const tgtMajor = parseInt(String(targetVersion).split('.')[0] || '0', 10);
    if (srcMajor !== tgtMajor) {
      return {
        compatible: false,
        class: 'feature-gap',
        reason: `patch-from-v${patchSourceVersion}-targets-different-major-version-v${targetVersion}`,
      };
    }
  }
  return { compatible: true };
}

/**
 * Conversion-axis check: can the installer's conversion helpers produce a coherent
 * target-runtime version of this patch?
 * Runs convertClaudeToCodexMarkdown (or converse) dry and inspects output for coherence.
 */
function conversionAxis({ patchContent, patchSourceRuntime, targetRuntime, relPath }) {
  // Same-runtime: no conversion needed
  if (patchSourceRuntime === targetRuntime) {
    return { compatible: true, converted: patchContent };
  }

  // Claude → Codex
  if (patchSourceRuntime === 'claude' && targetRuntime === 'codex') {
    try {
      let converted = patchContent;
      // Command-prefix + arg-sub + path-prefix conversions
      converted = convertClaudeToCodexMarkdown(converted);
      // Tool-name remap — apply by iterating the map
      for (const [claudeName, codexName] of Object.entries(claudeToCodexTools || {})) {
        if (codexName) {
          const re = new RegExp(`\\b${claudeName}\\b`, 'g');
          converted = converted.replace(re, codexName);
        }
      }
      // Path rewrite for agent specs
      if (relPath.startsWith('agents/')) {
        // use replacePathsInContent with known prefix replacements — match installer shape
        converted = replacePathsInContent(converted, '~/.claude/', '~/.codex/');
      }
      // If the converted output is structurally similar to the input but not identical → format-drift, convertible
      const sizeRatio = converted.length / (patchContent.length || 1);
      if (sizeRatio < 0.5 || sizeRatio > 2.0) {
        // Something went seriously wrong — conversion produced a degenerate output
        return {
          compatible: false,
          class: 'format-drift',
          reason: 'conversion-output-size-anomaly',
          low_confidence: true,
        };
      }
      return { compatible: true, converted };
    } catch (e) {
      return {
        compatible: false,
        class: 'format-drift',
        reason: `conversion-threw: ${e.message}`,
        low_confidence: true,
      };
    }
  }

  // Codex → Claude: reverse direction not yet fully supported (no inverse helper in install.js)
  // Mark as low-confidence; defer actual conversion to a later phase
  return {
    compatible: false,
    class: 'format-drift',
    reason: 'codex-to-claude-conversion-not-yet-implemented',
    low_confidence: true,
  };
}

/**
 * Main entry point: validate a patch file for applicability to the target runtime.
 *
 * @param {object} params
 * @param {string} params.patchContent - File contents of the patch
 * @param {'claude'|'codex'} params.patchSourceRuntime
 * @param {'claude'|'codex'} params.targetRuntime
 * @param {string} params.relPath - The patch's relative path within the runtime scope
 * @param {string} [params.patchSourceVersion]
 * @param {string} [params.targetVersion]
 * @param {string} [params.capabilityMatrixText]
 *
 * @returns {{
 *   compatible: boolean,
 *   class?: 'format-drift'|'feature-gap',
 *   confidence: 'high'|'medium'|'low',
 *   evidence: object,
 *   remediation: 'convert-and-apply'|'skip'|'abort'|null,
 *   low_confidence?: boolean,
 *   converted?: string,   // when compatible via conversion
 * }}
 */
function validatePatchForRuntime(params) {
  const { patchContent, patchSourceRuntime, targetRuntime, relPath } = params;
  if (!patchContent || !patchSourceRuntime || !targetRuntime || !relPath) {
    return {
      compatible: false,
      class: 'format-drift',
      confidence: 'low',
      evidence: { reason: 'invalid-validator-input' },
      remediation: 'abort',
      low_confidence: true,
    };
  }

  // Four-axis check in order
  const checks = [
    runtimeAxis(params),
    formatAxis(params),
    versionAxis(params),
    conversionAxis(params),
  ];

  // First incompatible result wins — subsequent checks don't override it
  for (const check of checks) {
    if (!check.compatible) {
      // Honor G-3: preserve low_confidence marker if any axis flagged it
      const lowConf = checks.some(c => c.low_confidence);
      return {
        compatible: false,
        class: check.class || 'format-drift',
        confidence: lowConf ? 'low' : 'medium',
        evidence: {
          axis: checks.indexOf(check),   // 0=runtime, 1=format, 2=version, 3=conversion
          reason: check.reason,
          relPath,
          source_runtime: patchSourceRuntime,
          target_runtime: targetRuntime,
        },
        remediation: check.class === 'feature-gap' ? 'skip' : (checks[3].converted ? 'convert-and-apply' : 'abort'),
        low_confidence: lowConf || undefined,
      };
    }
  }

  // All four axes passed — patch is compatible
  // If conversion-axis produced output, pass it through
  const convResult = checks[3];
  return {
    compatible: true,
    confidence: 'high',
    evidence: { all_axes_passed: true, relPath, source_runtime: patchSourceRuntime, target_runtime: targetRuntime },
    remediation: convResult.converted && convResult.converted !== patchContent ? 'convert-and-apply' : null,
    converted: convResult.converted,
  };
}

module.exports = {
  validatePatchForRuntime,
  runtimeAxis,
  formatAxis,
  versionAxis,
  conversionAxis,
  detectFormat,
};
```
