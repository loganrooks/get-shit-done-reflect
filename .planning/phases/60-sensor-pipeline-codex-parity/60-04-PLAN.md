```javascript
/**
 * Detect whether cwd is the GSDR source repo. Q2 closure.
 * Five-fold predicate — near-zero false-positive rate on live systems.
 * Returns: boolean
 */
function isDogfoodingRepo(cwd) {
  try {
    const pkgPath = path.join(cwd, 'package.json');
    if (!fs.existsSync(pkgPath)) return false;
    const pkg = JSON.parse(fs.readFileSync(pkgPath, 'utf8'));
    if (pkg.name !== 'get-shit-done-reflect-cc') return false;
    if (!fs.existsSync(path.join(cwd, '.git'))) return false;
    if (!fs.existsSync(path.join(cwd, 'bin', 'install.js'))) return false;
    if (!fs.existsSync(path.join(cwd, 'agents'))) return false;
    if (!fs.existsSync(path.join(cwd, 'get-shit-done'))) return false;
    return true;
  } catch {
    return false;
  }
}

/**
 * Load the installed capability-matrix.md for Q6 representability lookup.
 * Returns the markdown content string, or null if not found.
 * Looks under the runtime's install root (get-shit-done-reflect/references/capability-matrix.md).
 */
function loadCapabilityMatrix(runtimeDir) {
  try {
    const matrixPath = path.join(runtimeDir, 'get-shit-done-reflect', 'references', 'capability-matrix.md');
    if (fs.existsSync(matrixPath)) {
      return fs.readFileSync(matrixPath, 'utf8');
    }
  } catch { /* fall through */ }
  return null;
}

/**
 * Heuristic: does the artifact category (derived from relPath) have a surface on the target runtime?
 * Q6 closure: REPRESENTABILITY-based (not intentionality).
 * Returns: { applies: boolean, reason: string | null }
 */
function artifactCategoryApplies(relPath, runtime, capabilityMatrixText) {
  // If capability-matrix.md says the row for this artifact category is 'N' on the target runtime,
  // the category doesn't apply → feature-gap is the honest classification.
  // Simple heuristics for common categories:
  if (relPath.includes('hooks/') || relPath.endsWith('.hooks.json') || relPath.includes('settings.hooks')) {
    if (runtime === 'codex') {
      // Codex hooks are "under development" per Phase 58.1 DC-4 — does-not-apply-with-reason
      return { applies: false, reason: 'codex-hooks-under-development' };
    }
    return { applies: true, reason: null };
  }
  // For most categories (agents, workflows, references, commands/skills), both runtimes have surfaces.
  // Let the capability-matrix override by text search when needed.
  if (capabilityMatrixText) {
    // Lightweight heuristic: if the artifact's directory component is mentioned in a capability-matrix
    // row whose Codex column is exactly "N", the category doesn't apply.
    const dir = relPath.split('/')[0];
    const lines = capabilityMatrixText.split('\n').filter(l => l.startsWith('|') && l.toLowerCase().includes(dir.toLowerCase()));
    for (const line of lines) {
      const cells = line.split('|').map(c => c.trim());
      // Table shape is project-specific; look for a pattern: | category | ... | Claude | Codex | ...
      // If any cell referenceably in the Codex column is exactly 'N', representability is false.
      if (cells.some(c => c === 'N')) {
        // This is a rough heuristic — the cell's column is not strictly verified.
        // Good enough for Q6; refine if tests reveal false positives.
        return { applies: false, reason: `capability-matrix-row-marks-${runtime}-N` };
      }
    }
  }
  return { applies: true, reason: null };
}

/**
 * Classify a divergence entry.
 * Inputs per RESEARCH.md Pattern 3 skeleton.
 * Output: { class, confidence, evidence, severity, remediation, low_confidence? } | null (not-a-divergence)
 */
function classify({
  relPath,
  runtime,                      // 'claude' | 'codex'
  manifestHash,                 // SHA256 from gsd-file-manifest.json, or null
  installedHash,                // SHA256 of installed file, or null if missing
  sourceHash,                   // SHA256 of source-after-format-conversion, or null
  sourceFileExists,             // boolean — source still in agents/get-shit-done/commands?
  inLocalPatches,               // boolean — backed up to gsdr-local-patches/ or gsd-local-patches/?
  crossRuntimeInstalled,        // { claudeHash, codexHash } | null
  capabilityMatrixText,         // string content of capability-matrix.md, or null
  isDogfooding = false,
}) {
  // Layer 1: source-vs-installed
  if (manifestHash && installedHash && installedHash !== manifestHash) {
    if (sourceHash && installedHash === sourceHash) {
      // Installed matches source but not manifest → stale install (source changed after last install)
      return {
        class: 'stale',
        confidence: 'high',
        evidence: { manifestHash, installedHash, sourceHash, relPath, runtime },
        severity: isDogfooding ? 'trace' : 'notable',
        remediation: `node bin/install.js --${runtime}`,
      };
    }
    if (sourceHash && installedHash !== sourceHash && inLocalPatches) {
      // User modification preserved in local-patches/
      return {
        class: 'customization',
        confidence: 'high',
        evidence: { manifestHash, installedHash, sourceHash, relPath, runtime, inLocalPatches: true },
        severity: isDogfooding ? 'trace' : 'minor',
        remediation: '/gsdr:reapply-patches',
      };
    }
    if (!sourceFileExists) {
      // File in manifest + installed, but source gone — stale on renamed/deleted file
      return {
        class: 'stale',
        confidence: 'medium',
        evidence: { manifestHash, installedHash, sourceFileExists: false, relPath, runtime },
        severity: isDogfooding ? 'trace' : 'minor',
        remediation: `node bin/install.js --${runtime}  # removes orphans`,
      };
    }
    // Installer bug shape: hashes differ, no explanation
    return {
      class: 'bug',
      confidence: 'low',
      evidence: { manifestHash, installedHash, sourceHash, relPath, runtime },
      severity: isDogfooding ? 'trace' : 'notable',
      remediation: 'file a bug; check installer output for conversion errors',
      low_confidence: true,  // G-3: never silently downgrade
    };
  }

  // Layer 2: cross-runtime
  if (crossRuntimeInstalled && crossRuntimeInstalled.claudeHash && crossRuntimeInstalled.codexHash) {
    if (crossRuntimeInstalled.claudeHash !== crossRuntimeInstalled.codexHash) {
      // Format normalization must happen BEFORE declaring drift.
      // Caller is responsible for applying convertClaudeToCodexMarkdown / convertClaudeToCodexAgentToml
      // + replacePathsInContent to normalize; if hashes still differ post-normalization → format-drift.
      // This function ASSUMES the caller has already normalized if relevant.
      return {
        class: 'format-drift',
        confidence: 'medium',
        evidence: { ...crossRuntimeInstalled, relPath, runtime, note: 'post-normalization hash mismatch' },
        severity: isDogfooding ? 'trace' : 'minor',
        remediation: `node bin/install.js --${runtime === 'claude' ? 'codex' : 'claude'}`,
      };
    }
  }

  // Feature-gap: source exists, no installed-hash on target runtime, target runtime has NO surface (Q6)
  if (!installedHash && sourceFileExists) {
    const appl = artifactCategoryApplies(relPath, runtime, capabilityMatrixText);
    if (!appl.applies) {
      return {
        class: 'feature-gap',
        confidence: 'high',
        evidence: { runtime, relPath, sourceFileExists: true, representability_reason: appl.reason },
        severity: isDogfooding ? 'trace' : 'notable',
        remediation: null,  // no single-command remediation — capability doesn't apply on target runtime
      };
    }
    // Surface exists but isn't installed — that's a missing-install, not a feature-gap
    return {
      class: 'stale',
      confidence: 'medium',
      evidence: { runtime, relPath, sourceFileExists: true, reason: 'installer-did-not-include-file' },
      severity: isDogfooding ? 'trace' : 'minor',
      remediation: `node bin/install.js --${runtime}`,
    };
  }

  return null;  // no divergence
}

/**
 * Scan both patches directories (new + legacy names) — Pitfall 4.
 * Returns array of { filename, relPath, runtime, backupDir } for each candidate found.
 */
function scanPatchesDirectories(runtimeDir) {
  const results = [];
  for (const dirName of [PATCHES_DIR_NAME, LEGACY_PATCHES_DIR_NAME]) {
    const dir = path.join(runtimeDir, dirName);
    if (!fs.existsSync(dir)) continue;
    const metaFile = path.join(dir, 'backup-meta.json');
    if (!fs.existsSync(metaFile)) continue;
    try {
      const meta = JSON.parse(fs.readFileSync(metaFile, 'utf8'));
      if (meta.files && typeof meta.files === 'object') {
        for (const relPath of Object.keys(meta.files)) {
          results.push({
            filename: relPath,
            relPath,
            backupDir: dir,
            legacyNaming: (dirName === LEGACY_PATCHES_DIR_NAME),
            metaEntry: meta.files[relPath],
          });
        }
      }
    } catch { /* skip unreadable meta */ }
  }
  return results;
}
```

```javascript
function runSensor(cwd) {
  const isDogfooding = isDogfoodingRepo(cwd);
  const signals = [];
  const stats = { files_scanned: 0, divergences_found: 0, classification_failures: 0 };
  const runtimes = [
    { runtime: 'claude', dir: path.join(cwd, '.claude') },
    { runtime: 'codex', dir: path.join(cwd, '.codex') },
  ];
  for (const { runtime, dir } of runtimes) {
    if (!fs.existsSync(dir)) continue;
    const manifestPath = path.join(dir, MANIFEST_NAME);
    if (!fs.existsSync(manifestPath)) continue;
    let manifest;
    try {
      manifest = JSON.parse(fs.readFileSync(manifestPath, 'utf8'));
    } catch { stats.classification_failures += 1; continue; }
    const files = manifest.files || {};
    const capabilityMatrixText = loadCapabilityMatrix(dir);
    for (const [relPath, manifestEntry] of Object.entries(files)) {
      stats.files_scanned += 1;
      const installedPath = path.join(dir, relPath);
      const installedHash = fs.existsSync(installedPath) ? fileHash(installedPath) : null;
      const manifestHash = (typeof manifestEntry === 'object') ? manifestEntry.hash : manifestEntry;
      // Source-file existence check (rough — the source tree lives at cwd with different structure)
      // The installer's manifest already maps back via relPath; we check for existence of
      // the parallel source file by stripping the runtime-scope prefix.
      const sourceRelPath = relPath.startsWith('get-shit-done-reflect/') ? relPath.substring('get-shit-done-reflect/'.length) : relPath;
      const sourcePath = path.join(cwd, sourceRelPath);
      const sourceFileExists = fs.existsSync(sourcePath);
      const sourceHash = null;  // TODO: compute source-after-format-conversion hash for high-confidence classifications.
      //   For Phase 60 v1, we leave sourceHash null — the classifier degrades to medium-confidence stale for this case.
      //   Plan 60-06's XRT-02 validator may upgrade this.
      const result = classify({
        relPath, runtime, manifestHash, installedHash, sourceHash,
        sourceFileExists, inLocalPatches: false,  // Layer-1 only for sensor's initial pass
        crossRuntimeInstalled: null,
        capabilityMatrixText, isDogfooding,
      });
      if (result) {
        stats.divergences_found += 1;
        signals.push({
          signal_type: (result.class === 'feature-gap' || result.class === 'format-drift') ? 'capability-gap' : 'convention-deviation',
          severity: result.severity,
          tags: ['patch-sensor', result.class, `runtime-${runtime}`].concat(result.low_confidence ? ['low-confidence'] : []),
          evidence: result.evidence,
          remediation: result.remediation,
          low_confidence: !!result.low_confidence,
          detected_by: { runtime: 'unknown', sensor: 'gsd-patch-sensor' },  // harness fills this in if it can
          about_work: [{ path: relPath, runtime, kind: 'installed-file' }],
        });
      }
    }
  }
  return { signals, stats, dogfooding_mode: isDogfooding };
}
```
