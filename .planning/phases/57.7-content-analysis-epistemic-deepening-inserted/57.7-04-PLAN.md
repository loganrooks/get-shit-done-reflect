```javascript
/**
 * Scan assistant records for tool_use blocks and build an ordered tool-name sequence.
 * Returns ONLY tool names, never arguments (DC-3 / content_contract: derived_features_only).
 * @param {Array} records - parsed JSONL records from session.parent_jsonl.records
 * @returns {{ sequence: string[], tool_count_per_tool: Object<string,number>, distinct_tool_count: number, tool_sequence_entropy: number }}
 */
function scanToolInvocationSequence(records) {
  const sequence = [];
  const tool_count_per_tool = {};
  for (const record of records || []) {
    if (!record || record.type !== 'assistant') continue;
    const content = record.message && record.message.content;
    if (!Array.isArray(content)) continue;
    for (const block of content) {
      // Guard per RESEARCH.md Pitfall 2 — never iterate without type check
      if (!block || block.type !== 'tool_use' || typeof block.name !== 'string') continue;
      sequence.push(block.name);
      tool_count_per_tool[block.name] = (tool_count_per_tool[block.name] || 0) + 1;
    }
  }
  const distinct = Object.keys(tool_count_per_tool);
  const total = sequence.length;
  let entropy = 0;
  for (const name of distinct) {
    const p = tool_count_per_tool[name] / total;
    entropy -= p * Math.log2(p);
  }
  return {
    sequence,
    tool_count_per_tool,
    distinct_tool_count: distinct.length,
    tool_sequence_entropy: total > 0 ? entropy : 0,
  };
}

/**
 * Trivial tool-category grouping for topic-shift detection.
 * Categories: 'investigation' (Read/Grep/Glob), 'mutation' (Edit/Write/MultiEdit), 'execution' (Bash).
 * Unknown tools map to 'other'. Keep minimal per RESEARCH.md Q-New-2 "Don't Hand-Roll".
 */
function toolCategoryFor(name) {
  if (['Read', 'Grep', 'Glob', 'WebFetch'].includes(name)) return 'investigation';
  if (['Edit', 'Write', 'MultiEdit'].includes(name)) return 'mutation';
  if (['Bash'].includes(name)) return 'execution';
  return 'other';
}

/**
 * Scan for STRUCTURAL topic-shift markers. No semantic analysis (G-1).
 * Counts: (a) /clear invocations, (b) compaction boundaries, (c) tool-category changes
 * between consecutive assistant tool-use blocks, (d) turn-size discontinuity
 * (record output-token count > 2σ from preceding rolling mean) — IF output_tokens available.
 * Returns positions and counts only — never record content.
 */
function scanTopicShiftMarkers(records) {
  const clearCount = countClearInvocations(records);
  const compaction = extractCompactionEvents(records);
  const shift_positions = [];

  // (c) tool-category changes between adjacent tool_use blocks
  let lastCategory = null;
  let tool_category_shift_count = 0;
  for (let i = 0; i < (records || []).length; i++) {
    const record = records[i];
    if (!record || record.type !== 'assistant') continue;
    const content = record.message && record.message.content;
    if (!Array.isArray(content)) continue;
    for (const block of content) {
      if (!block || block.type !== 'tool_use' || typeof block.name !== 'string') continue;
      const cat = toolCategoryFor(block.name);
      if (lastCategory !== null && cat !== lastCategory) {
        tool_category_shift_count++;
        shift_positions.push(i);
      }
      lastCategory = cat;
    }
  }

  const topic_shift_count = clearCount + compaction.count + tool_category_shift_count;
  const total_turns = (records || []).length;
  const mean_shift_interval_turns = topic_shift_count > 1 && total_turns > 0
    ? total_turns / topic_shift_count
    : null;

  return {
    topic_shift_count,
    clear_shifts: clearCount,
    compaction_shifts: compaction.count,
    tool_category_shifts: tool_category_shift_count,
    mean_shift_interval_turns,
    shift_positions,
  };
}

// Export — append to existing module.exports block at file bottom
module.exports = {
  // ... existing exports preserved ...
  scanToolInvocationSequence,
  scanTopicShiftMarkers,
  toolCategoryFor,
};
```
