**Step A — Add `suggestDistinguishingFeatures`:**

```javascript
/**
 * Registry-traversal distinguishing-feature suggester (MEAS-ARCH-09).
 * Scans the registry for extractors whose `distinguishes` metadata overlaps
 * the interpretation's declared distinguishing_features AND that have NOT
 * been computed in the current feature set.
 *
 * NOT LLM-assisted per G-4 / trial-before-formalize. Deterministic and testable.
 *
 * @param {Object} interpretation - interpretation object with distinguishing_features array
 * @param {Array} registryExtractors - registry.extractors list
 * @param {Array} computedFeatures - features actually computed for the current scope
 * @returns {Array<{extractor, distinguishes, reason}>}
 */
function suggestDistinguishingFeatures(interpretation, registryExtractors, computedFeatures) {
  const interpDistinguishes = new Set(
    // interpretation.distinguishing_features is an array of feature names (strings) in 57.5/57.6
    // In 57.7 we may see richer shapes {feature, symmetry_markers}; normalize to strings
    (interpretation.distinguishing_features || []).map(d =>
      typeof d === 'string' ? d : (d && d.feature) || ''
    ).filter(Boolean)
  );
  if (interpDistinguishes.size === 0) return [];

  const computedExtractorNames = new Set(
    (computedFeatures || []).map(f => f.extractor).filter(Boolean)
  );

  const suggestions = [];
  for (const extractor of registryExtractors || []) {
    if (!extractor.distinguishes || !Array.isArray(extractor.distinguishes)) continue;
    const overlap = extractor.distinguishes.filter(d => interpDistinguishes.has(d));
    if (overlap.length === 0) continue;
    if (computedExtractorNames.has(extractor.name)) continue; // already computed
    suggestions.push({
      extractor: extractor.name,
      distinguishes: overlap,
      reason: 'registered_but_not_computed_for_current_scope',
    });
  }
  return suggestions;
}
```

**Step B — Add `loadInterventionOutcomes` using `extractFrontmatter`:**

```javascript
const path = require('node:path');
const fs = require('node:fs');

/**
 * Reads side-car intervention-outcome records from .planning/measurement/interventions/
 * Filters by interpretation_id via exact string equality. Returns an array of outcome
 * records joined to the interpretation.
 *
 * File-first source of truth per DC-10 resolution-on-demand.
 * Uses extractFrontmatter from get-shit-done/bin/lib/frontmatter.cjs (canonical utility).
 */
function loadInterventionOutcomes(cwd, interpretationId) {
  if (!interpretationId) return [];
  const dir = path.join(cwd, '.planning', 'measurement', 'interventions');
  if (!fs.existsSync(dir)) return [];
  if (!fs.statSync(dir).isDirectory()) return [];
  const files = fs.readdirSync(dir).filter(f => f.endsWith('.md') && f !== 'README.md');
  const outcomes = [];
  for (const file of files) {
    const fullPath = path.join(dir, file);
    let content;
    try {
      content = fs.readFileSync(fullPath, 'utf8');
    } catch (_) {
      continue;
    }
    const record = extractFrontmatter(content);
    if (!record || typeof record !== 'object') continue;
    if (record.interpretation_id !== interpretationId) continue;
    outcomes.push({
      intervention_id: record.intervention_id || null,
      intervention_artifact: record.intervention_artifact || null,
      predicted_outcome: record.predicted_outcome || null,
      actual_outcome: record.actual_outcome || null,
      outcome_status: record.outcome_status || 'pending',
      evaluation_date: record.evaluation_date || null,
      record_path: path.relative(cwd, fullPath),
    });
  }
  return outcomes;
}
```

**Step C — Add `aggregateRevisionHistory` (also using `extractFrontmatter`):**

```javascript
/**
 * Aggregates revision_classification distribution from diagnostic frontmatter.
 * Reads .planning/measurement/diagnostics/*.md; for each diagnostic matching
 * the interpretation_id (or that declares `revises: <interpretation_id>`),
 * collects the `revision_classification` field.
 *
 * Returns both the list of revisions and a distribution count.
 */
function aggregateRevisionHistory(cwd, interpretationId) {
  if (!interpretationId) return { revisions: [], distribution: {} };
  const dir = path.join(cwd, '.planning', 'measurement', 'diagnostics');
  if (!fs.existsSync(dir)) return { revisions: [], distribution: {} };
  if (!fs.statSync(dir).isDirectory()) return { revisions: [], distribution: {} };
  const files = fs.readdirSync(dir).filter(f => f.endsWith('.md') && f !== 'README.md');
  const revisions = [];
  const distribution = {};
  for (const file of files) {
    const fullPath = path.join(dir, file);
    let content;
    try {
      content = fs.readFileSync(fullPath, 'utf8');
    } catch (_) {
      continue;
    }
    const record = extractFrontmatter(content);
    if (!record) continue;
    // Match: diagnostic whose id matches the interpretation, OR diagnostic that declares
    // `revises: <interpretationId>` in frontmatter (link-back to revised artifact).
    const matches = record.interpretation_id === interpretationId
      || record.id === interpretationId
      || record.diagnostic_id === interpretationId
      || record.revises === interpretationId
      || (Array.isArray(record.revises) && record.revises.includes(interpretationId));
    if (!matches) continue;
    const classification = record.revision_classification || 'unclassified';
    revisions.push({
      diagnostic_path: path.relative(cwd, fullPath),
      timestamp: record.revised_at || record.created_at || null,
      classification,
      justification: record.revision_justification || null,
    });
    distribution[classification] = (distribution[classification] || 0) + 1;
  }
  return { revisions, distribution };
}
```

**Step D — Add `composeProvenanceSummary`:**

```javascript
/**
 * Composes the full-provenance summary string on query (DC-10).
 * Governed wording per DC-7 / G-governing: never uses 'verified' or 'confirmed to be true'.
 */
function composeProvenanceSummary(interpretation, interventionOutcomes, revisionDistribution, anomalyCount) {
  const challengeCount = Array.isArray(interpretation.competing_readings)
    ? interpretation.competing_readings.length
    : 0;
  const confirmed = (interventionOutcomes || []).filter(i => i.outcome_status === 'confirmed').length;
  const mixed = (interventionOutcomes || []).filter(i => i.outcome_status === 'mixed').length;
  const grounded = confirmed + mixed;
  const pending = (interventionOutcomes || []).filter(i => i.outcome_status === 'pending').length;

  const revisionKeys = Object.keys(revisionDistribution || {});
  const revisionSummary = revisionKeys.length === 0
    ? 'no_revisions'
    : revisionKeys
        .sort()
        .map(k => `${k}=${revisionDistribution[k]}`)
        .join(', ');

  // Governed wording — pending is disclosed, grounded is the strongest phrasing
  const parts = [
    `surviving_challenge_from_${challengeCount}_perspectives`,
    `grounded_in_${grounded}_confirmed_or_mixed_interventions`,
  ];
  if (pending > 0) parts.push(`${pending}_pending_interventions`);
  parts.push(`carrying_${anomalyCount}_tracked_anomalies`);
  parts.push(`revisions=[${revisionSummary}]`);
  return parts.join(' | ');
}
```
