const reasoningQualityProxyExtractor = defineExtractor({
  name: 'reasoning_quality_proxy',
  source_family: 'DERIVED',
  raw_sources: ['facets_semantic_summary'],  // chained derivation
  runtimes: ['claude-code'],  // facets is currently Claude-only
  reliability_tier: 'inferred',   // DC-4 / G-5 — NON-NEGOTIABLE
  features_produced: ['reasoning_quality_proxy'],
  serves_loop: ['agent_performance'],
  distinguishes: ['reasoning_quality_signal', 'facets_backed_quality_inference'],
  status_semantics: ['exposed', 'not_emitted', 'not_available'],
  extract(extractor, context) {
    // Read facets_semantic_summary features from the current feature set
    const allFeatures = context.computedFeatures || context.features || [];
    const facetsFeatures = allFeatures.filter(f => f.extractor === 'facets_semantic_summary');
    const results = [];
    for (const facet of facetsFeatures) {
      const sessionId = facet.value && facet.value.session_id;
      if (!sessionId) continue;
      const available = facet.availability_status === 'exposed' && facet.value;
      if (!available) {
        results.push(buildFeatureRecord(extractor, {
          feature_name: `reasoning_quality_proxy:${sessionId}`,
          runtime: 'claude-code',
          availability_status: 'not_available',
          skip_reason: facet.skip_reason || 'facets_unavailable',
          reliability_tier: 'inferred',
          value: null,
          coverage: facet.coverage || {},
          provenance: {
            session_id: sessionId,
            source_feature: 'facets_semantic_summary',
            proxy_label: 'reasoning_quality_proxy_only',
          },
          freshness: facet.freshness || {},
        }));
        continue;
      }
      // Compose proxy score from facets fields (per Plan 02 DECISION methodology)
      const v = facet.value;
      // Baseline scoring per Plan 02 trial: goal presence + primary_success + inverse friction + helpfulness
      let score = 0;
      if (v.underlying_goal) score += 1;
      if (v.primary_success) score += 1;
      if (typeof v.claude_helpfulness === 'number') score += Math.max(0, Math.min(1, v.claude_helpfulness));
      const friction = (v.friction_counts && typeof v.friction_counts.total === 'number') ? v.friction_counts.total : 0;
      score -= Math.min(1, friction / 10);   // cap negative contribution
      const normalized = Math.max(0, Math.min(1, score / 3));   // 0..1 range
      results.push(buildFeatureRecord(extractor, {
        feature_name: `reasoning_quality_proxy:${sessionId}`,
        runtime: 'claude-code',
        availability_status: 'exposed',
        reliability_tier: 'inferred',
        value: {
          session_id: sessionId,
          reasoning_quality_proxy_score: normalized,
          proxy_mechanism: 'facets-substitute',
          proxy_label: 'reasoning_quality_proxy_only',
          component_signals: {
            has_underlying_goal: !!v.underlying_goal,
            primary_success: !!v.primary_success,
            claude_helpfulness: v.claude_helpfulness,
            friction_count_total: friction,
          },
        },
        coverage: facet.coverage || {},
        provenance: {
          session_id: sessionId,
          source_feature: 'facets_semantic_summary',
          mechanism: 'facets-substitute',
          decision_ref: '.planning/spikes/012-C5-reasoning-quality-mechanism/DECISION.md',
        },
        freshness: facet.freshness || {},
        notes: [
          'G-5 / DC-4: this is a proxy, not a reasoning-quality measurement.',
          'Spike 010 Finding #6 (summary length not a proxy) governs labelling.',
          `Mechanism: facets-substitute per Plan 02 DECISION ${v.underlying_goal ? 'with goal' : 'without goal'}.`,
        ],
      }));
    }
    return results;
  },
});

const REFERENCE_PATTERNS = [
  /@\\.planning\\//g,
  /@get-shit-done\\//g,
  /\\bphase\\s+5[0-9]\\b/gi,
  /\\bsig-[0-9]{4}-/g,
  /\\bspk-[0-9]{4}-/g,
  /[a-z0-9_-]+\\.md\\b/gi,
];

function computeReferenceDensity(records) {
  let reference_hits = 0;
  let total_chars = 0;
  for (const record of records || []) {
    if (!record || record.type !== 'assistant') continue;
    const content = record.message && record.message.content;
    if (!Array.isArray(content)) continue;
    for (const block of content) {
      if (block && block.type === 'text' && typeof block.text === 'string') {
        total_chars += block.text.length;
        for (const re of REFERENCE_PATTERNS) {
          const matches = block.text.match(re);
          if (matches) reference_hits += matches.length;
        }
      }
    }
  }
  // Normalize per ~1k tokens (char-count ÷ 4 heuristic unless Plan 08 gave us a tokenizer)
  const approx_tokens = Math.max(1, total_chars / 4);
  return {
    reference_hits,
    approx_tokens,
    references_per_1k_tokens: (reference_hits * 1000) / approx_tokens,
  };
}
