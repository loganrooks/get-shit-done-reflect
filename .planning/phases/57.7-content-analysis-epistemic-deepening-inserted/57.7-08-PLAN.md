```javascript
/**
 * Read token_count.reasoning_output_tokens from a Codex rollout record.
 * Returns numeric count or null if the field is absent (Pitfall 4 guard).
 * @param {Object} record - a Codex rollout record with token_count
 * @returns {number | null}
 */
function readReasoningOutputTokens(record) {
  if (!record || typeof record !== 'object') return null;
  const tc = record.token_count;
  if (!tc || typeof tc !== 'object') return null;
  const value = tc.reasoning_output_tokens;
  if (typeof value !== 'number' || !Number.isFinite(value) || value < 0) return null;
  return value;
}

/**
 * Summarize Codex session reasoning tokens.
 * Sums reasoning_output_tokens across all rollout records in a session.
 * Returns {reasoning_tokens: number | null, field_present: boolean}.
 * field_present: false when the field is absent from ALL records → drives skip_reason.
 */
function summarizeCodexReasoningTokens(session) {
  if (!session || !session.rollout || !Array.isArray(session.rollout.records)) {
    return { reasoning_tokens: null, field_present: false };
  }
  let anyPresent = false;
  let total = 0;
  for (const record of session.rollout.records) {
    const value = readReasoningOutputTokens(record);
    if (value !== null) {
      anyPresent = true;
      total += value;
    }
  }
  return {
    reasoning_tokens: anyPresent ? total : null,
    field_present: anyPresent,
  };
}

module.exports = {
  // ... existing exports preserved ...
  readReasoningOutputTokens,
  summarizeCodexReasoningTokens,
};
```

```javascript
function computeSessionReasoningDelta(records) {
  let total_output = 0;
  let total_visible = 0;
  let total_thinking = 0;
  let negative = false;

  for (const record of records || []) {
    if (!record || record.type !== 'assistant') continue;
    const msg = record.message;
    if (!msg) continue;
    const output_tokens = msg.usage && typeof msg.usage.output_tokens === 'number' ? msg.usage.output_tokens : 0;
    const content = Array.isArray(msg.content) ? msg.content : [];
    let visible_text = '';
    let thinking_text = '';
    for (const block of content) {
      if (!block) continue;
      if (block.type === 'text' && typeof block.text === 'string') visible_text += block.text;
      if (block.type === 'thinking' && typeof block.thinking === 'string') thinking_text += block.thinking;
      // tool_use blocks are visible output but not text — they're accounted for in output_tokens
      // by Anthropic; we count their serialized names only (best-effort approximation)
      if (block.type === 'tool_use' && typeof block.name === 'string') visible_text += block.name;
    }
    const visibleCount = countTokens(visible_text);
    const thinkingCount = countTokens(thinking_text);
    const delta = output_tokens - visibleCount - thinkingCount;
    total_output += output_tokens;
    total_visible += visibleCount;
    total_thinking += thinkingCount;
    if (delta < 0) negative = true;
  }
  const reasoning_tokens = total_output - total_visible - total_thinking;
  return {
    reasoning_tokens,
    negative_delta_flag: negative || reasoning_tokens < 0,
  };
}
```
