```bash
# XRT-01: planning-phase assertion — hook-dependent commitments require Codex path
CONTEXT_FILE="$phase_dir/${padded_phase}-CONTEXT.md"
if [ -f "$CONTEXT_FILE" ]; then
  # Heuristic: does CONTEXT mention hook-dependent keywords?
  HAS_HOOK_COMMITMENT=$(grep -ciE 'hook|SessionStop|postlude|codex_hooks|session-meta' "$CONTEXT_FILE" || true)

  if [ "$HAS_HOOK_COMMITMENT" -gt 0 ]; then
    # Does CONTEXT contain an explicit Codex degradation section?
    # Accept any of: "Codex behavior", "does-not-apply-with-reason", "applies-via-workflow-step if codex_hooks",
    #                "Codex degradation", "waiver path", section referencing XRT-01
    HAS_CODEX_PATH=$(grep -ciE 'Codex behavior|does-not-apply-with-reason|applies-via-workflow-step|Codex degradation|waiver path|XRT-01' "$CONTEXT_FILE" || true)

    if [ "$HAS_CODEX_PATH" -eq 0 ]; then
      echo "::notice::gate_fired=XRT-01 result=block reason=hook_commitment_no_codex_path"
      echo "XRT-01 blocks planning: CONTEXT.md mentions hook-dependent features but lacks explicit Codex degradation / waiver path."
      echo "Required: add a Codex behavior section declaring applies / applies-via-workflow-step / does-not-apply-with-reason for each hook-dependent commitment."
      echo "See 58-05-codex-behavior-matrix.md for the standard vocabulary."
      exit 1
    fi
    echo "::notice::gate_fired=XRT-01 result=pass reason=codex_path_declared"
  else
    echo "::notice::gate_fired=XRT-01 result=pass reason=no_hook_commitments"
  fi
fi
```

Heuristic caveat: a false positive is possible (CONTEXT mentions "hook" casually). Document in HEADNOTE. The `HAS_CODEX_PATH` OR-condition is permissive; anyone authoring CONTEXT in a hook-dependent phase can easily satisfy it by adding a small Codex-behavior block.
