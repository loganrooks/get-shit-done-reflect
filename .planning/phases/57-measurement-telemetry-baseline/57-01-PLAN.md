const fs = require('fs');

**Helpers to implement:**

1. `normalizeProjectPath(projectPath)` -- strips `/.claude/worktrees/{name}` suffix from project_path to get the main project root. Use regex: `/\/\.claude\/worktrees\/[^/]+$/`.

2. `loadSessionMeta(options, homeDir)` -- reads all JSON files from `{homeDir}/.claude/usage-data/session-meta/`. Parameters:
   - `options`: `{ project, since, until }` (all optional)
   - `homeDir`: defaults to `os.homedir()` (injectable for testing)
   - Wraps each `JSON.parse()` in try/catch, counts skipped files
   - If `options.project` is set, normalize both the filter path (via `resolveWorktreeRoot(options.project)`) and each session's `project_path` (via `normalizeProjectPath()`), then filter
   - Returns `{ sessions: [], skipped: N, total: N }`

3. `loadFacets(homeDir)` -- reads all JSON files from `{homeDir}/.claude/usage-data/facets/`. Returns a `Map<session_id, facetsData>`. Same try/catch with skip counting. `homeDir` defaults to `os.homedir()`.

4. `filterByDateRange(sessions, since, until)` -- filters sessions by `start_time` field.
   - Parse `since` as start-of-day UTC: `new Date(since + 'T00:00:00Z')`
   - Parse `until` as end-of-day UTC: `new Date(until + 'T23:59:59.999Z')`
   - If date is invalid (isNaN), call `error('Invalid date format: ...')`.
   - Filter by `new Date(session.start_time).getTime()` within range.

5. `computePercentiles(values)` -- takes an array of numbers, returns `{ min, p25, median, p75, p90, max, n }` or null if empty. Sort, use linear interpolation for percentile indices. Implementation per RESEARCH.md Pattern 4.

6. `enrichSession(session, facetsMap)` -- merges facets data with a session record. If facets exist for the session_id, adds a `quality` object with outcome, satisfaction, friction_types, session_type, and `_annotation: "AI-generated estimate with unknown accuracy"`. If no facets, sets `quality: null`.

7. `formatDistribution(label, stats, unit)` -- formats a percentile distribution as a fixed-width table row (for human-readable output). Right-align numbers with `String.padStart()`, left-align label with `String.padEnd()`.

8. `formatSummaryTable(sessions, facetsMap)` -- formats the human-readable summary output showing session count, date range, project filter status, skipped count, and top-level distribution stats for duration and tokens.

**Commands to implement:**

1. `cmdTelemetrySummary(cwd, options, raw, homeDir)` -- `options`: `{ project, since, until }`. Loads session-meta with project/date filtering. Computes summary stats: total sessions, date range (earliest start_time to latest), duration distribution, token distributions (input/output), facets coverage count. Human-readable default shows a table; `--raw` returns JSON.

2. `cmdTelemetrySession(cwd, sessionId, raw, homeDir)` -- loads a single session-meta file by sessionId (filename = `{sessionId}.json`). Loads facets and enriches. If session not found, `error('Session not found: ...')`. Human-readable shows key-value pairs; `--raw` returns the full enriched JSON.

3. `cmdTelemetryEnrich(cwd, sessionId, raw, homeDir)` -- similar to session but focused on the facets join. Loads session-meta + facets, calls `enrichSession()`, outputs enriched record. Every facets-derived field carries the `_annotation` per TEL-05.

**Exports:** All cmd* functions plus helpers (loadSessionMeta, loadFacets, computePercentiles, normalizeProjectPath, filterByDateRange, enrichSession) for testing.

**Key constraints:**
- All functions are synchronous (per established lib module pattern).
- `homeDir` parameter on every data-loading function defaults to `os.homedir()` -- this is the testing seam.
- No JSONL reading in this plan (JSONL is only for the token validation task in Plan 02).
- Facets annotation string must be exactly: `"AI-generated estimate with unknown accuracy"` per TEL-05.

**2. Add router case (after the `case 'kb':` block, before the `default:` case):**
```javascript
case 'telemetry': {
  const subcommand = args[1];
  if (subcommand === 'summary') {
    const projectIdx = args.indexOf('--project');
    const sinceIdx = args.indexOf('--since');
    const untilIdx = args.indexOf('--until');
    const options = {
      project: projectIdx !== -1 ? args[projectIdx + 1] : undefined,
      since: sinceIdx !== -1 ? args[sinceIdx + 1] : undefined,
      until: untilIdx !== -1 ? args[untilIdx + 1] : undefined,
    };
    telemetry.cmdTelemetrySummary(cwd, options, raw);
  } else if (subcommand === 'session') {
    telemetry.cmdTelemetrySession(cwd, args[2], raw);
  } else if (subcommand === 'enrich') {
    telemetry.cmdTelemetryEnrich(cwd, args[2], raw);
  } else if (subcommand === 'phase') {
    telemetry.cmdTelemetryPhase(cwd, args[2], raw);
  } else if (subcommand === 'baseline') {
    const projectIdx = args.indexOf('--project');
    const options = {
      project: projectIdx !== -1 ? args[projectIdx + 1] : undefined,
    };
    telemetry.cmdTelemetryBaseline(cwd, options, raw);
  } else {
    error('Unknown telemetry subcommand. Available: summary, session, phase, baseline, enrich');
  }
  break;
}
```
