  const args = ['telemetry', subcommand, ...extraArgs, '--raw']
  const result = execSync(`node "${GSD_TOOLS}" ${args.join(' ')}`, {
    cwd: tmpdir,
    env: { ...process.env, HOME: path.join(tmpdir, 'home') },
    encoding: 'utf-8',
    timeout: 15000,
    stdio: ['pipe', 'pipe', 'pipe'],
  })
  return JSON.parse(result.trim())
}
